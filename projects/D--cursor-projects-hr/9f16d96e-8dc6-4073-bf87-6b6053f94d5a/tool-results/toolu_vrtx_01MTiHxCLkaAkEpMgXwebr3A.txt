  1190→        suffix = '?',

  1195→    try {
  1196→        if (sufIndex === -1) {
  1197→            res = str.substr(preIndex + prefix.length)
  1198→        } else {
  1199→            res = str.substr(preIndex + prefix.length, sufIndex - preIndex - prefix.length)
  1200→        }
  1201→
  1202→        return res
  1203→    } catch (err) {

  1206→    }
