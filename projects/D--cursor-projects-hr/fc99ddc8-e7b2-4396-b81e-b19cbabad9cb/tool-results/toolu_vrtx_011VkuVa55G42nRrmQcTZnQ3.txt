   169→    public function addWrittenExamInfo(Request $request)
   170→    {
   171→        db()->beginTransaction();
   172→        $status = $request->status ?? '';
   173→        $is_archive = $request->is_archive ?? false; // 1表示归档0表示不归档
   174→        try {
   175→            if ($status == 'fail' && $is_archive) { // 如果笔试未通过
   176→                $resumeId = RecruitmentProcessionMd::find($request->rec_pro_id)->resume_id;
   177→                $resume = ResumeMd::where('id', $resumeId)->first();
   178→                $this->service->updateProgress(
   179→                    '笔试未通过',
   180→                    $resumeId,
   181→                    $resume->recent_apply_job_id
   182→                );
   183→                // 更新简历表
   184→                $updateResData = $this->service->getResumeUpdateInfo('fail', '', '笔试未通过', $request->remark ?? '', $request->rec_pro_id);
   185→                $this->service->resumeUpdateUseRecId($updateResData, $request->rec_pro_id);
   186→                // 更新招聘流程表
   187→                $this->service->recruitmentUpdate(['status' => RecruitmentProcessionMd::REC_STATUS['归档']], $request->rec_pro_id);
   188→            }
   189→            $addFlag = empty($request->id);
   190→//            $old_status = $addFlag ? null : WrittenExamMd::find($request->id)->status;
   191→            if (!$addFlag) {
   192→                $columns = ['evaluation', 'status', 'remark', 'exam_status', 'total_score'];
   193→                $changeColumns = [];
   194→                $old = WrittenExamMd::find($request->id);
   195→                foreach ($columns as $v) {
   196→                    $new = $request->{$v} ?? '';
   197→                    if ($new && $new != $old->{$v}) {
   198→                        $changeColumns[$v] = ['old' => $old->{$v}, 'new' => $new];
   199→                    }
   200→                }
   201→                $subject = $request->subject ?? '';
   202→                $oldSubject = Arr::get($old->exam_content, 'subject', '');
   203→                if ($subject && $subject != $oldSubject) {
   204→                    $changeColumns['subject'] = ['old' => $oldSubject, 'new' => $subject];
   205→                }
   206→            }
   207→            $addArray = ['status' => $status];
   208→            if (!$addFlag) {
   209→                $exam_content = WrittenExamMd::find($request->id)->exam_content;
   210→            }
   211→            $subject = $request->subject ?? '';
   212→            if ($subject) {
   213→                $exam_content['subject'] = $request->subject;
   214→                $addArray['exam_content'] = $exam_content;
   215→            }
   216→            $this->createOrUpdateWrittenExam($request->all() + $addArray);
   217→            $rec = RecruitmentProcessionMd::find($request->rec_pro_id);
   218→            if (!$addFlag && !empty($changeColumns)) {
   219→                ResumeOperationMd::create(
   220→                    [
   221→                        'resume_id' => $rec->resume_id ?? 0,
   222→                        'cur_status' => $rec->status,
   223→                        'type' => 'written_exam_changed',
   224→                        'extra_info' => ['rec_pro_id' => $rec->id, 'change_data' => $changeColumns, 'is_smb' => 0],
   225→                        'reason' => '笔试修改',
   226→                        'job_id' => $rec->job_id,
   227→                        'operator_id' => AuthRecord::user()->user_id,
   228→                        'specific_reason' => $request->remark ?? $request->evaluation ?? '',
   229→                    ]
   230→                );
   231→            }
   232→            if ($addFlag) {
   233→                ResumeOperationMd::create(
   234→                    [
   235→                        'resume_id' => $rec->resume_id ?? 0,
   236→                        'cur_status' => $rec->status,
   237→                        'type' => 'add_written_exam_result',
   238→                        'extra_info' => ['rec_pro_id' => $rec->id, 'is_smb' => 0],
   239→                        'reason' => '添加笔试结果',
   240→                        'job_id' => $rec->job_id,
   241→                        'operator_id' => AuthRecord::user()->user_id,
   242→                        'specific_reason' => $request->remark ?? $request->evaluation ?? '',
   243→                    ]
   244→                );
   245→            }
   246→            // 记录归档操作
   247→            if ($is_archive) {
   248→                ResumeOperationMd::create(
   249→                    [
   250→                        'resume_id' => $rec->resume_id ?? 0,
   251→                        'cur_status' => $rec->status,
   252→                        'type' => 'archive',
   253→                        'extra_info' => ['rec_pro_id' => $rec->id, 'is_smb' => 0],
   254→                        'reason' => '添加笔试结果并归档',
   255→                        'job_id' => $rec->job_id,
   256→                        'operator_id' => AuthRecord::user()->user_id,
   257→                        'specific_reason' => $request->remark ?? $request->evaluation ?? '',
   258→                    ]
   259→                );
   260→            }
   261→
   262→            if ($status == 'fail' && $is_archive) { // 如果笔试未通过
   263→                // 网申自动判断再次转入初筛
   264→                $this->service->ifOnlineAutoNextJob([$request->rec_pro_id]);
   265→            }
   266→
   267→            db()->commit();
   268→        } catch (\Exception $e) {
   269→            db()->rollBack();
   270→            return $this->responseJsonFail(__('my_api.update_written_exam_info_fail').$e->getMessage(), 424);
   271→        }
   272→
   273→        return $this->responseJsonSuccess([]);
   274→    }

   344→    /**
   345→     * 转入复筛并发送邮件给复筛人.
   346→     *
   347→     * @param Request $request
   348→     * @return JsonResponse
   349→     */
   350→    public function toRescreening(Request $request)
   351→    {
   352→        $input = $request->all();
   353→        $this->validateRescreening($request->all());
   354→        $resumeIds = array_column($input['recruitments'], 'resume_id');
   355→        $recruitmentIds = array_column($input['recruitments'], 'rec_pro_id');
   356→        $specificReason = array_column($input['recruitments'], 'specific_reason');
   357→
   358→        db()->beginTransaction();
   359→        try {
   360→            $msg = '';
   361→            if ($input['result'] != 'pass') {
   362→                $resume_id = $resumeIds[0];
   363→                $resume = ResumeMd::where('id', $resume_id)->first();
   364→                $msg = $resume->status == RecruitmentProcessionMd::REC_STATUS['初筛'] ? '初筛未通过' : '笔试未通过';
   365→            }
   366→
   367→            foreach ($input['recruitments'] as $rec) {
   368→                $this->service->singleToScreening(array_merge($rec, Arr::only($input, ['result', 'screener_id'])));
   369→            }
   370→
   371→            if ($input['result'] == 'pass') { // 发送复筛邮件
   372→                setAuthorityToken($input['authorized_token'], $resumeIds, $input['screener_id'], 1728000);
   373→                $this->mail_service->sendResumeFilterEmail($recruitmentIds, userInfo()[$input['screener_id']]['nickname'], $specificReason[0] ?? '', $input['authorized_token']);
   374→            } else {
   375→                $this->service->ifOnlineAutoNextJob($recruitmentIds, $msg);
   376→            }
   377→            db()->commit();
   378→        } catch (\Exception $e) {
   379→            db()->rollBack();
   380→            return $this->responseJsonFail(__('my_api.to_rescreen_fail').$e->getMessage(), 423);
   381→        }
   382→        if ($input['result'] != 'pass' && $input['is_send_rejection_letter'] == true) {
   383→            // 初筛没有通过，发送拒信邮件
   384→            foreach ($recruitmentIds as $id) {
   385→                $this->mail_service->sendRefuseEmail($id, $input['authorized_token']);
   386→            }
   387→        }
   388→        return $this->responseJsonSuccess([]);
   389→    }

   391→    /**
   392→     * 在复筛中页面，重新发送复筛邮件，需要判断候选人状态
   393→     *
   394→     * @param Request $request
   395→     * @return JsonResponse
   396→     */
   397→    public function resendResumeFilterEmail(Request $request)
   398→    {
   399→        $screenerId = $request->screener_id;
   400→        $recProIds = $request->rec_pro_ids;
   401→        $legalId = RecruitmentProcessionMd::whereIn('id', $recProIds)->where('status', '=', RecruitmentProcessionMd::REC_STATUS['复筛'])->pluck('resume_id', 'id')->toArray();
   402→        $resumeId = array_values($legalId);
   403→        $legalId = array_keys($legalId);
   404→        ResumeFilterMd::whereIn('rec_pro_id', $legalId)->where('round_times', 2)->update(['updated_at' => date('Y-m-d H:i:s')]);
   405→        $illegalId = array_diff($recProIds, $legalId);
   406→        if (!empty($illegalId)) {
   407→            return $this->responseJsonFail(__('my_api.resend_resume_filter_mail_fail', ['illegalId' => implode(',', $illegalId), 'recProIds' => implode(',', $recProIds)]), UserException::ERR_USER_OPERATE_LIMIT);
   408→        }
   409→        setAuthorityToken($request->authorized_token, $resumeId, $screenerId, 1728000);
   410→        $this->mail_service->sendResumeFilterEmail($recProIds, userInfo()[$screenerId]['nickname'], '', $request->authorized_token);
   411→        return $this->responseJsonSuccess([]);
