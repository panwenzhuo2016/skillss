+    public void mmWebhookFeedback(MmFeedbackWebhookRo ro) {

+        String extraInfo = buildExtraInfo(ro.getMessageId());

+    private void mmHandleUpdate(MmFeedbackWebhookRo ro) {

+            mmHandleCreate(ro);
+            return;
+        }
+
+        List<String> attachmentUrls = downloadMmFiles(ro);
+        existing.setContent(ro.getContent());
+        existing.setAttachments(attachmentUrls.isEmpty() ? null : JSONUtil.toJsonStr(attachmentUrls));
+        existing.setFeedbackTime(parseFeedbackTime(ro.getFeedbackTime()));
+        existing.setExtraInfo(buildExtraInfo(ro.getMessageId()));
+        feedbackMapper.updateById(existing);
+        log.info("[lbl][2087-消息推送][Feedback][UPDATE] 更新成功, id={}, postId={}", existing.getId(), ro.getPostId());

+    private void mmHandleDelete(MmFeedbackWebhookRo ro) {
+        FeedbackEntity existing = findByPostId(ro.getPostId());
+        if (existing == null) {
+            log.warn("[lbl][2087-消息推送][Feedback][DELETE] postId={} 记录不存在, 忽略", ro.getPostId());
+            return;

+        existing.setIsDeleted(1);
+        feedbackMapper.updateById(existing);
+        log.info("[lbl][2087-消息推送][Feedback][DELETE] 逻辑删除成功, id={}, postId={}", existing.getId(), ro.getPostId());
+    }

+    private FeedbackEntity findByPostId(String postId) {

+    @Override
+    public IPage<FeedbackVo> listFeedback(FeedbackListRo ro) {
+        int pageNum = ro.getPageNum() != null ? ro.getPageNum() : 1;
+        int pageSize = ro.getPageSize() != null ? ro.getPageSize() : 20;
+
+        LambdaQueryWrapper<FeedbackEntity> wrapper = new LambdaQueryWrapper<>();
+        wrapper.eq(FeedbackEntity::getIsDeleted, 0);
+
+        if (ro.getStatus() != null) {
+            wrapper.eq(FeedbackEntity::getStatus, ro.getStatus());
+        }
+        if (StrUtil.isNotBlank(ro.getDocType())) {
+            wrapper.eq(FeedbackEntity::getDocType, ro.getDocType());
+        }
+        if (StrUtil.isNotBlank(ro.getFeedbackType())) {
+            wrapper.eq(FeedbackEntity::getFeedbackType, ro.getFeedbackType());
+        }
+        if (StrUtil.isNotBlank(ro.getStartDate())) {
+            try {
+                LocalDate startDate = LocalDate.parse(ro.getStartDate(), DateTimeFormatter.ISO_LOCAL_DATE);
+                wrapper.ge(FeedbackEntity::getFeedbackTime, startDate.atStartOfDay());
+            } catch (java.time.format.DateTimeParseException e) {
+                throw new BusinessException("开始日期格式错误，应为yyyy-MM-dd");
+            }
+        }
+        if (StrUtil.isNotBlank(ro.getEndDate())) {
+            try {
+                LocalDate endDate = LocalDate.parse(ro.getEndDate(), DateTimeFormatter.ISO_LOCAL_DATE);
+                wrapper.le(FeedbackEntity::getFeedbackTime, endDate.plusDays(1).atStartOfDay());
+            } catch (java.time.format.DateTimeParseException e) {
+                throw new BusinessException("结束日期格式错误，应为yyyy-MM-dd");
+            }
+        }
+        if (StrUtil.isNotBlank(ro.getKeyword())) {
+            wrapper.like(FeedbackEntity::getContent, ro.getKeyword());
+        }
+
+        wrapper.orderByDesc(FeedbackEntity::getFeedbackTime);
+
+        IPage<FeedbackEntity> page = feedbackMapper.selectPage(new Page<>(pageNum, pageSize), wrapper);
+
+        return page.convert(this::toVo);
+    }

+    private List<String> uploadFilesToMinio(List<MultipartFile> files) {
+        List<String> urls = new ArrayList<>();
+        if (files == null || files.isEmpty()) {
+            return urls;
+        }
+        String bucketName = constProperties.getOssBucketByAsset().getBucketName();
+        for (MultipartFile file : files) {
+            try {
+                String keyPath = "feedback/" + LocalDate.now().format(DateTimeFormatter.BASIC_ISO_DATE)
+                        + "/" + UUID.randomUUID() + "_" + file.getOriginalFilename();
+                ossTemplate.upload(bucketName, file.getInputStream(), keyPath);
+                String url = constProperties.getOssBucketByAsset().getResourceUrl() + "/" + keyPath;
+                urls.add(url);
+            } catch (IOException e) {
+                log.error("[lbl][2087-消息推送][Feedback][UPLOAD] 附件上传MinIO失败, fileName={}, error={}", file.getOriginalFilename(), e.getMessage(), e);
+            }
+        }
+        return urls;
+    }

+                log.info("[lbl][2087-消息推送][Feedback][DOWNLOAD] MM附件下载完成, fileId={}, fileSize={}bytes", fileId, fileData.length);
+                String keyPath = "feedback/mm/" + LocalDate.now().format(DateTimeFormatter.BASIC_ISO_DATE)
+                        + "/" + UUID.randomUUID() + "_" + fileId;
+                ossTemplate.upload(bucketName, new ByteArrayInputStream(fileData), keyPath);
+                String url = constProperties.getOssBucketByAsset().getResourceUrl() + "/" + keyPath;
+                urls.add(url);
+                log.info("[lbl][2087-消息推送][Feedback][UPLOAD] MM附件上传MinIO成功, fileId={}, keyPath={}", fileId, keyPath);
+            } catch (Exception e) {
+                log.error("[lbl][2087-消息推送][Feedback][UPLOAD] MM附件下载上传失败, fileId={}, error={}", fileId, e.getMessage(), e);
+            }
+        }
