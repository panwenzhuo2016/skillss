   178→    public void handleWebhookMergedBatch(MergedBatch mergedBatch) {
   179→        if (mergedBatch == null) {
   180→            log.warn("mergedBatch为空，跳过处理");
   181→            return;
   182→        }
   183→        OperationSourceHolder.set(OperationSourceHolder.NEXTCLOUD_WEBHOOK);
   184→        try {
   185→            log.info("开始处理mergedBatch. deletes: {}, renames: {}, creates: {}",
   186→                    CollectionUtils.size(mergedBatch.deletes()),
   187→                    CollectionUtils.size(mergedBatch.renames()),
   188→                    CollectionUtils.size(mergedBatch.creates()));
   189→            handleDeletedTasks(mergedBatch.deletes());
   190→            handleRenamedTasks(mergedBatch.renames());
   191→            handleCreatedTasks(mergedBatch.creates());
   192→            log.info("mergedBatch处理完成");
   193→        } finally {
   194→            OperationSourceHolder.clear();
   195→        }
   196→    }

   198→    public void syncNodeWrittenUpdateTime(String webhookPayload) throws JsonProcessingException {
   199→        NextcloudNodeWrittenEvent nodeWrittenEvent = objectMapper.readValue(webhookPayload, NextcloudNodeWrittenEvent.class);
   200→        if (nodeWrittenEvent == null || nodeWrittenEvent.getEvent() == null || nodeWrittenEvent.getEvent().getNode() == null) {
   201→            log.warn("Nextcloud webhook nodeWritten payload invalid");
   202→            return;
   203→        }
   204→        Integer fileId = nodeWrittenEvent.getEvent().getNode().getId();
   205→        if (fileId == null) {
   206→            log.warn("Nextcloud webhook nodeWritten missing fileId");
   207→            return;
   208→        }
   209→        Long eventTime = nodeWrittenEvent.getTime();
   210→        if (eventTime == null || eventTime <= 0) {
   211→            log.warn("Nextcloud webhook nodeWritten missing or invalid updateTime, fileId={}", fileId);
   212→            return;
   213→        }
   214→        LocalDateTime updateDateTime = LocalDateTime.ofInstant(Instant.ofEpochSecond(eventTime), ZoneId.systemDefault());
   215→        String updateTime = UPDATE_TIME_FORMATTER.format(updateDateTime);
   216→        String params = objectMapper.writeValueAsString(buildUploadUpdateTimeParamMap(String.valueOf(fileId), updateTime));
   217→        thirdAppUploadDataServiceFactory.getService(SpaceNodeTypeEnum.NEXTCLOUD.getValue()).uploadUpdateTime(params);
   218→    }

   227→    private void handleDeletedTasks(List<NextcloudWebhookTaskDto> tasks) {
   228→        if (CollectionUtils.isEmpty(tasks)) {
   229→            log.info("删除任务为空，跳过处理");
   230→            return;
   231→        }
   232→        log.info("开始处理删除任务. taskCount: {}", tasks.size());
   233→        TaskStats stats = new TaskStats();
   234→        Map<String, List<NextcloudWebhookTaskDto>> groupedByUserName = tasks.stream()
   235→                .filter(Objects::nonNull)
   236→                .filter(task -> StringUtils.isNotBlank(task.getPath()))
   237→                .collect(Collectors.groupingBy(task -> StringUtils.defaultString(task.getUserName())));
   238→        log.info("删除任务分组完成. userGroupCount: {}", groupedByUserName.size());
   239→        for (Map.Entry<String, List<NextcloudWebhookTaskDto>> entry : groupedByUserName.entrySet()) {
   240→            String userName = entry.getKey();
   241→            if (StringUtils.isBlank(userName)) {
   242→                log.warn("删除任务存在空userName分组，已跳过. groupTaskCount: {}", entry.getValue().size());
   243→                continue;
   244→            }
   245→            Long userId = setUserHolder(userName);
   246→            if (userId == null) {
   247→                log.warn("删除任务设置用户上下文失败，已跳过该分组. userName: {}, groupTaskCount: {}", userName, entry.getValue().size());
   248→                continue;
   249→            }
   250→            log.info("开始处理删除任务分组. userName: {}, userId: {}, groupTaskCount: {}", userName, userId, entry.getValue().size());
   251→            for (NextcloudWebhookTaskDto task : entry.getValue()) {
   252→                try {
   253→                    ParsedNextcloudPath parsedPath = parseNextcloudPath(task.getPath());
   254→                    if (parsedPath == null) {
   255→                        log.warn("删除任务路径解析失败，已跳过. fileId: {}, path: {}", task.getFileId(), task.getPath());
   256→                        stats.markSkipped();
   257→                        continue;
   258→                    }
   259→                    SpaceNodeEntity existedSpaceNode = findExistedSpaceNodeByPath(parsedPath.spaceId(), parsedPath.folders(), parsedPath.fileName(), task.getFileId());
   260→                    if (existedSpaceNode == null) {
   261→                        log.info("删除任务未找到对应节点，已跳过. fileId: {}, path: {}", task.getFileId(), task.getPath());
   262→                        stats.markSkipped();
   263→                        continue;
   264→                    }
   265→                    log.info("执行删除任务. userName: {}, spaceNodeId: {}, fileId: {}, path: {}",
   266→                            userName, existedSpaceNode.getId(), task.getFileId(), task.getPath());
   267→                    // 通知 rag-doc-server 删除 Ragflow 中对应的文档
   268→                    if (ragDocNotifyPublisher != null) {
   269→                        ragDocNotifyPublisher.publishDocDelete(existedSpaceNode);
   270→                    }
   271→                    spaceNodeUpdateManager.deleteSpaceNode(existedSpaceNode.getId(), userId);
   272→                    stats.markSuccess();
   273→                } catch (Exception e) {
   274→                    stats.markFailed();
   275→                    log.error("删除原子任务处理失败，已隔离. userName: {}, fileId: {}, path: {}, error: {}",
   276→                            userName, task.getFileId(), task.getPath(), e.getMessage(), e);
   277→                }
   278→            }
   279→        }
   280→        log.info("删除任务处理完成. successCount: {}, skippedCount: {}, failedCount: {}",
   281→                stats.success, stats.skipped, stats.failed);
   282→    }

   326→    private void handleCreatedTasks(List<NextcloudWebhookTaskDto> tasks) {
   327→        if (CollectionUtils.isEmpty(tasks)) {
   328→            log.info("创建任务为空，跳过处理");
   329→            return;
   330→        }
   331→        log.info("开始处理创建任务. taskCount: {}", tasks.size());
   332→        TaskStats stats = new TaskStats();
   333→        Map<String, List<NextcloudWebhookTaskDto>> groupedByUserName = tasks.stream()
   334→                .filter(Objects::nonNull)
   335→                .filter(task -> StringUtils.isNotBlank(task.getPath()))
   336→                .collect(Collectors.groupingBy(task -> StringUtils.defaultString(task.getUserName())));
   337→        log.info("创建任务按用户分组完成. userGroupCount: {}", groupedByUserName.size());
   338→        for (Map.Entry<String, List<NextcloudWebhookTaskDto>> userEntry : groupedByUserName.entrySet()) {
   339→            String userName = userEntry.getKey();
   340→            if (StringUtils.isBlank(userName)) {
   341→                log.warn("创建任务存在空userName分组，已跳过. groupTaskCount: {}", userEntry.getValue().size());
   342→                continue;
   343→            }
   344→            Long userId = setUserHolder(userName);
   345→            if (userId == null) {
   346→                log.warn("创建任务设置用户上下文失败，已跳过该分组. userName: {}, groupTaskCount: {}", userName, userEntry.getValue().size());
   347→                continue;
   348→            }
   349→            log.info("开始处理创建任务用户分组. userName: {}, userId: {}, groupTaskCount: {}", userName, userId, userEntry.getValue().size());
   350→            Map<String, List<NextcloudWebhookTaskDto>> groupedBySpaceId = userEntry.getValue().stream()
   351→                    .collect(Collectors.groupingBy(task -> {
   352→                        ParsedNextcloudPath parsed = parseNextcloudPath(task.getPath());
   353→                        return parsed == null ? "" : parsed.spaceId();
   354→                    }));
   355→            log.info("创建任务按space分组完成. userName: {}, spaceGroupCount: {}", userName, groupedBySpaceId.size());
   356→            for (Map.Entry<String, List<NextcloudWebhookTaskDto>> spaceEntry : groupedBySpaceId.entrySet()) {
   357→                String spaceId = spaceEntry.getKey();
   358→                if (StringUtils.isBlank(spaceId)) {
   359→                    log.warn("创建任务存在空spaceId分组，已跳过. userName: {}, groupTaskCount: {}", userName, spaceEntry.getValue().size());
   360→                    continue;
   361→                }
   362→                List<NextcloudWebhookTaskDto> spaceTasks = spaceEntry.getValue();
   363→                log.info("开始处理创建任务分组. userName: {}, spaceId: {}, groupTaskCount: {}", userName, spaceId, spaceTasks.size());
   364→                Set<String> existingFileIds = loadExistingFileIds(spaceId, spaceTasks);
   365→                Map<String, Boolean> pathTypeMap = preloadDirectoryTypeMap(spaceTasks);
   366→                List<CreateItem> createItems = new ArrayList<>();
   367→                for (NextcloudWebhookTaskDto task : spaceTasks) {
   368→                    try {
   369→                        ParsedNextcloudPath parsed = parseNextcloudPath(task.getPath());
   370→                        if (parsed == null || StringUtils.isBlank(parsed.fileName())) {
   371→                            log.warn("创建任务路径解析失败或文件名为空，已跳过. userName: {}, fileId: {}, path: {}", userName, task.getFileId(), task.getPath());
   372→                            stats.markSkipped();
   373→                            continue;
   374→                        }
   375→                        String relativePath = extractRelativePathFromWebhookPath(task.getPath());
   376→                        if (StringUtils.isBlank(relativePath)) {
   377→                            relativePath = buildRelativePath(parsed);
   378→                        }
   379→                        boolean isFolder = Boolean.TRUE.equals(pathTypeMap.get(relativePath));
   380→                        Integer nodeType = isFolder ? SpaceNodeTypeEnum.FOLDER.getValue() : SpaceNodeTypeEnum.NEXTCLOUD.getValue();
   381→                        String detail = isFolder ? buildFolderCreateDetail(relativePath) : buildFileCreateDetail(relativePath);
   382→
   383→                        boolean marked = false;
   384→                        if (StringUtils.isNotBlank(task.getFileId())) {
   385→                            marked = markerService.isMarkedAndConsume(parsed.spaceId(), task.getFileId(), NextcloudSyncOperationEnum.CREATE, detail);
   386→                        }
   387→                        if (!marked) {
   388→                            marked = markerService.isMarkedAndConsumeByPath(parsed.spaceId(), NextcloudSyncOperationEnum.CREATE, detail);
   389→                        }
   390→                        if (marked) {
   391→                            log.info("创建任务命中自同步标记，跳过. userName: {}, spaceId: {}, fileId: {}, path: {}",
   392→                                    userName, parsed.spaceId(), task.getFileId(), task.getPath());
   393→                            stats.markSkipped();
   394→                            continue;
   395→                        }
   396→
   397→                        if (StringUtils.isNotBlank(task.getFileId()) && existingFileIds.contains(task.getFileId())) {
   398→                            log.info("创建任务节点已存在，已跳过. userName: {}, fileId: {}, path: {}", userName, task.getFileId(), task.getPath());
   399→                            stats.markSkipped();
   400→                            continue;
   401→                        }
   402→
   403→                        if (isFolder) {
   404→                            SpaceNodeEntity existedFolder = findExistedSpaceNodeByPath(parsed.spaceId(), parsed.folders(), parsed.fileName(), null);
   405→                            if (existedFolder != null) {
   406→                                log.info("创建任务文件夹同名已存在，跳过. userName: {}, spaceId: {}, folderName: {}, path: {}",
   407→                                        userName, parsed.spaceId(), parsed.fileName(), task.getPath());
   408→                                stats.markSkipped();
   409→                                continue;
   410→                            }
   411→                        }
   412→
   413→                        createItems.add(new CreateItem(parsed.folders(), parsed.fileName(), nodeType, task.getFileId()));
   414→                    } catch (Exception e) {
   415→                        stats.markFailed();
   416→                        log.error("创建原子任务预处理失败，已隔离. userName: {}, fileId: {}, path: {}, error: {}",
   417→                                userName, task.getFileId(), task.getPath(), e.getMessage(), e);
   418→                    }
   419→                }
   420→                List<CreateBucket> buckets = nextcloudCreateTreeBuilder.buildBuckets(createItems);
   421→                log.info("创建任务构建bucket完成. userName: {}, spaceId: {}, createItemCount: {}, bucketCount: {}",
   422→                        userName, spaceId, createItems.size(), buckets.size());
   423→                for (CreateBucket bucket : buckets) {
   424→                    Long parentNodeId = ensureFoldersExist(spaceId, bucket.parentFolders());
   425→                    if (parentNodeId == null) {
   426→                        log.warn("创建任务bucket父目录准备失败，已跳过该bucket. userName: {}, spaceId: {}, parentFolders: {}, rootCount: {}",
   427→                                userName, spaceId, bucket.parentFolders(), CollectionUtils.size(bucket.roots()));
   428→                        continue;
   429→                    }
   430→                    List<ResolvedCreateRoot> resolvedRoots = resolveBucketRootConflicts(spaceId, parentNodeId, bucket.roots());
   431→                    for (ResolvedCreateRoot resolvedRoot : resolvedRoots) {
   432→                        SpaceNodeBatchCreateReq root = resolvedRoot.root();
   433→                        try {
   434→                            log.info("执行创建原子任务. userName: {}, spaceId: {}, parentNodeId: {}, rootName: {}, rootType: {}",
   435→                                    userName, spaceId, parentNodeId, root.getName(), root.getType());
   436→                            List<TreeNode<SpaceNodeVo>> createdRoots =
   437→                                    spaceNodeUpdateManager.createSpaceNodesOnlyInAioBatch(spaceId, parentNodeId, List.of(root));
   438→                            syncResolvedCreateRootNameIfNeeded(createdRoots, resolvedRoot, spaceId, bucket.parentFolders(), userId);
   439→                            // 通知 RAG 索引新创建的 NEXTCLOUD 文件节点

   445→                                    userName, spaceId, parentNodeId, root.getName(), e.getMessage(), e);
   446→                        }
   447→                    }
   448→                }
   449→            }
   450→        }
   451→        log.info("创建任务处理完成. successCount: {}, skippedCount: {}, failedCount: {}",
   452→                stats.success, stats.skipped, stats.failed);
   453→    }

   557→    private Map<String, Boolean> preloadDirectoryTypeMap(List<NextcloudWebhookTaskDto> tasks) {
   558→        if (CollectionUtils.isEmpty(tasks)) {
   559→            return Map.of();
   560→        }
   561→        // relativePath -> task, 保留 task 引用以便 fallback 时使用 fileId/spaceName
   562→        Map<String, NextcloudWebhookTaskDto> pathToTask = new LinkedHashMap<>();
   563→        for (NextcloudWebhookTaskDto task : tasks) {
   564→            if (task == null || StringUtils.isBlank(task.getPath())) {
   565→                continue;
   566→            }
   567→            String relativePath = extractRelativePathFromWebhookPath(task.getPath());
   568→            if (StringUtils.isBlank(relativePath)) {
   569→                continue;
   570→            }
   571→            pathToTask.putIfAbsent(relativePath, task);
   572→        }
   573→        if (pathToTask.isEmpty()) {
   574→            return Map.of();
   575→        }
   576→
   577→        Set<String> relativePaths = pathToTask.keySet();
   578→        Map<String, Boolean> pathTypeMap = new ConcurrentHashMap<>(relativePaths.size());
   579→        Set<String> parentPaths = collectParentPaths(relativePaths);
   580→        List<String> unresolvedPaths = new ArrayList<>();
   581→        for (String relativePath : relativePaths) {
   582→            if (parentPaths.contains(relativePath)) {
   583→                pathTypeMap.put(relativePath, true);
   584→                continue;
   585→            }
   586→            if (isLikelyFileByExtension(relativePath)) {
   587→                pathTypeMap.put(relativePath, false);
   588→                continue;
   589→            }
   590→            unresolvedPaths.add(relativePath);
   591→        }
   592→        if (unresolvedPaths.isEmpty()) {
   593→            return pathTypeMap;
   594→        }
   595→
   596→        log.info("创建任务路径类型预判完成. totalPathCount: {}, folderByTreeCount: {}, fileByExtCount: {}, unresolvedCount: {}",
   597→                relativePaths.size(),
   598→                parentPaths.stream().filter(relativePaths::contains).count(),
   599→                pathTypeMap.values().stream().filter(Boolean.FALSE::equals).count(),
   600→                unresolvedPaths.size());
   601→
   602→        try (ExecutorService executor = Executors.newVirtualThreadPerTaskExecutor()) {
   603→            List<CompletableFuture<Void>> futures = new ArrayList<>(unresolvedPaths.size());
   604→            for (String relativePath : unresolvedPaths) {
   605→                NextcloudWebhookTaskDto task = pathToTask.get(relativePath);
   606→                CompletableFuture<Void> future = CompletableFuture.runAsync(() -> {
   607→                    boolean isFolder = isDirectoryInNextcloud(relativePath, task.getFileId(), task.getSpaceName());
   608→                    pathTypeMap.put(relativePath, isFolder);
   609→                }, executor).exceptionally(ex -> {
   610→                    log.warn("预取Nextcloud路径类型失败，默认按文件夹处理. path: {}, error: {}", relativePath, ex.getMessage());
   611→                    pathTypeMap.put(relativePath, true);
   612→                    return null;
   613→                });
   614→                futures.add(future);
   615→            }
   616→            CompletableFuture.allOf(futures.toArray(new CompletableFuture[0])).join();
   617→        }
   618→        return pathTypeMap;
   619→    }

   621→    private Set<String> collectParentPaths(Set<String> relativePaths) {
   622→        Set<String> parentPaths = new HashSet<>();
   623→        for (String relativePath : relativePaths) {
   624→            if (StringUtils.isBlank(relativePath)) {
   625→                continue;
   626→            }
   627→            int index = relativePath.lastIndexOf('/');
   628→            while (index > 0) {
   629→                parentPaths.add(relativePath.substring(0, index));
   630→                index = relativePath.lastIndexOf('/', index - 1);
   631→            }
   632→        }
   633→        return parentPaths;
   634→    }

   636→    private boolean isLikelyFileByExtension(String relativePath) {
   637→        if (StringUtils.isBlank(relativePath)) {
   638→            return false;
   639→        }
   640→        int slashIndex = relativePath.lastIndexOf('/');
   641→        String fileName = slashIndex >= 0 ? relativePath.substring(slashIndex + 1) : relativePath;
   642→        if (StringUtils.isBlank(fileName)) {
   643→            return false;
   644→        }
   645→        int dotIndex = fileName.lastIndexOf('.');
   646→        if (dotIndex <= 0 || dotIndex == fileName.length() - 1) {
   647→            return false;
   648→        }
   649→        String extension = fileName.substring(dotIndex + 1).toLowerCase();
   650→        return COMMON_FILE_EXTENSIONS.contains(extension);
   651→    }

   653→    private String extractRelativePathFromWebhookPath(String fieldPath) {
   654→        if (StringUtils.isBlank(fieldPath)) {
   655→            return null;
   656→        }
   657→        String[] pathParts = fieldPath.split("/");
   658→        if (pathParts.length < MIN_PATH_PARTS) {
   659→            return null;
   660→        }
   661→        int spaceNameIndex = -1;
   662→        for (int i = 0; i < pathParts.length; i++) {
   663→            if (NEXTCLOUD_FILES_MARKER.equals(pathParts[i]) && i + 1 < pathParts.length) {
   664→                spaceNameIndex = i + 1;
   665→                break;
   666→            }
   667→        }
   668→        if (spaceNameIndex == -1 || spaceNameIndex >= pathParts.length) {
   669→            return null;
   670→        }
   671→        StringBuilder relativePath = new StringBuilder();
   672→        for (int i = spaceNameIndex; i < pathParts.length; i++) {
   673→            String part = pathParts[i];
   674→            if (StringUtils.isBlank(part)) {
   675→                continue;
   676→            }
   677→            if (!relativePath.isEmpty()) {
   678→                relativePath.append('/');
   679→            }
   680→            relativePath.append(part);
   681→        }
   682→        if (relativePath.isEmpty()) {
   683→            return null;
   684→        }
   685→        return relativePath.toString();
   686→    }

   743→    private TaskProcessResult processRenamedTask(NextcloudWebhookTaskDto task, String userName, Long userId) {
   744→        String sourcePath = resolveRenameSourcePath(task);
   745→        String targetPath = task.getPath();
   746→        if (StringUtils.isBlank(sourcePath) || StringUtils.isBlank(targetPath)) {
   747→            log.warn("重命名任务source/target路径缺失，已跳过. fileId: {}, sourcePath: {}, targetPath: {}",
   748→                    task.getFileId(), sourcePath, targetPath);
   749→            return TaskProcessResult.SKIPPED;
   750→        }
   751→
   752→        ParsedNextcloudPath source = parseNextcloudPath(sourcePath);
   753→        ParsedNextcloudPath target = parseNextcloudPath(targetPath);
   754→        if (source == null || target == null || !Objects.equals(source.spaceId(), target.spaceId())) {
   755→            log.warn("重命名任务路径解析失败或跨space，已跳过. fileId: {}, sourcePath: {}, targetPath: {}",
   756→                    task.getFileId(), sourcePath, targetPath);
   757→            return TaskProcessResult.SKIPPED;
   758→        }
   759→
   760→        String sourceRelativePath = buildRelativePath(source);
   761→        String targetRelativePath = buildRelativePath(target);
   762→        String moveDetail = buildFileMoveDetail(sourceRelativePath, targetRelativePath);
   763→        if (isRenameSyncMarked(target.spaceId(), task.getFileId(), moveDetail)) {
   764→            log.info("重命名任务命中自同步标记，已跳过. userName: {}, fileId: {}, sourcePath: {}, targetPath: {}",
   765→                    userName, task.getFileId(), sourcePath, targetPath);
   766→            return TaskProcessResult.SKIPPED;
   767→        }
   768→
   769→        SpaceNodeEntity existedSpaceNode = findExistedSpaceNodeByPath(source.spaceId(), source.folders(), source.fileName(), task.getFileId());
   770→        if (existedSpaceNode == null) {
   771→            log.info("重命名任务未找到源节点，已跳过. fileId: {}, sourcePath: {}", task.getFileId(), sourcePath);
   772→            return TaskProcessResult.SKIPPED;
   773→        }
   774→
   775→        NodeEntity currentNode = nodeService.getByRelId(existedSpaceNode.getId());
   776→        if (currentNode == null) {
   777→            log.warn("重命名任务无法获取当前节点，已跳过. spaceNodeId: {}, fileId: {}",
   778→                    existedSpaceNode.getId(), task.getFileId());
   779→            return TaskProcessResult.SKIPPED;
   780→        }
   781→
   782→        boolean sameParent = Objects.equals(source.folders(), target.folders());
   783→        Long targetParentNodeId = currentNode.getParentId();
   784→        if (!sameParent) {
   785→            targetParentNodeId = ensureFoldersExist(target.spaceId(), target.folders());
   786→            if (targetParentNodeId == null) {
   787→                log.warn("重命名任务目标父目录准备失败，已跳过. fileId: {}, targetPath: {}", task.getFileId(), targetPath);
   788→                return TaskProcessResult.SKIPPED;
   789→            }
   790→        }
   791→
   792→        String resolvedTargetName = resolveAvailableTargetName(target.spaceId(), targetParentNodeId, target.fileName(),
   793→                existedSpaceNode.getId(), existedSpaceNode.getType());
   794→        if (StringUtils.isBlank(resolvedTargetName)) {
   795→            log.warn("重命名任务无法解析最终目标名，已跳过. fileId: {}, targetPath: {}", task.getFileId(), targetPath);
   796→            return TaskProcessResult.SKIPPED;
   797→        }
   798→
   799→        if (!sameParent) {
   800→            String temporaryMoveName = resolveTemporaryMoveName(target.spaceId(), currentNode.getParentId(), targetParentNodeId, existedSpaceNode);
   801→            if (!Objects.equals(temporaryMoveName, existedSpaceNode.getName())) {
   802→                log.info("执行重命名任务中的临时改名. userName: {}, spaceNodeId: {}, tempName: {}, fileId: {}",
   803→                        userName, existedSpaceNode.getId(), temporaryMoveName, task.getFileId());
   804→                handleRenameOnly(existedSpaceNode, temporaryMoveName, userId);
   805→                existedSpaceNode = refreshSpaceNode(existedSpaceNode.getId());
   806→            }
   807→
   808→            MoveSpaceNodeRo moveRo = new MoveSpaceNodeRo();
   809→            moveRo.setNodeId(currentNode.getId());
   810→            moveRo.setNewParentId(targetParentNodeId);
   811→            moveRo.setNewPreId(ROOT_PRE_ID);
   812→            log.info("执行重命名任务中的移动. userName: {}, spaceId: {}, nodeId: {}, targetParentNodeId: {}, sourcePath: {}, targetPath: {}",
   813→                    userName, target.spaceId(), currentNode.getId(), targetParentNodeId, sourcePath, targetPath);
   814→            spaceNodeUpdateManager.moveSpaceNode(target.spaceId(), moveRo);
   815→            existedSpaceNode = refreshSpaceNode(existedSpaceNode.getId());
   816→        }
   817→
   818→        if (existedSpaceNode == null) {
   819→            log.warn("重命名任务移动后节点不存在，已跳过. fileId: {}, targetPath: {}", task.getFileId(), targetPath);
   820→            return TaskProcessResult.SKIPPED;
   821→        }
   822→        boolean needSyncBack = !Objects.equals(resolvedTargetName, target.fileName());
   823→        if (Objects.equals(existedSpaceNode.getName(), resolvedTargetName)) {
   824→            log.info("重命名任务无需再次改名. userName: {}, spaceNodeId: {}, resolvedName: {}, fileId: {}",
   825→                    userName, existedSpaceNode.getId(), resolvedTargetName, task.getFileId());
   826→            if (needSyncBack) {
   827→                syncResolvedConflictName(existedSpaceNode, resolvedTargetName, target, userId);
   828→            }
   829→            return TaskProcessResult.SUCCESS;
   830→        }
   831→        log.info("执行重命名任务中的改名. userName: {}, spaceNodeId: {}, requestedName: {}, resolvedName: {}, fileId: {}, needSyncBack: {}",
   832→                userName, existedSpaceNode.getId(), target.fileName(), resolvedTargetName, task.getFileId(), needSyncBack);
   833→        applyResolvedRename(existedSpaceNode, resolvedTargetName, target, userId, needSyncBack);
   834→        return TaskProcessResult.SUCCESS;
   835→    }

   872→    private Set<String> loadOccupiedNames(String spaceId, List<Long> parentNodeIds, Long excludeSpaceNodeId) {
   873→        Set<String> occupiedNames = new HashSet<>();
   874→        Set<Long> uniqueParentIds = new LinkedHashSet<>(parentNodeIds);
   875→        for (Long parentNodeId : uniqueParentIds) {
   876→            if (parentNodeId == null) {
   877→                continue;
   878→            }
   879→            List<NodeEntity> childNodes = nodeService.getByTreeIdAndParentIdAndNodeKey(spaceId, parentNodeId, NodeKeyType.AIO_FILE_TREE.getValue());
   880→            if (CollectionUtils.isEmpty(childNodes)) {
   881→                continue;
   882→            }
   883→            List<Long> relIds = childNodes.stream()
   884→                    .map(NodeEntity::getRelId)
   885→                    .filter(Objects::nonNull)
   886→                    .filter(relId -> !Objects.equals(relId, excludeSpaceNodeId))
   887→                    .toList();
   888→            if (CollectionUtils.isEmpty(relIds)) {
   889→                continue;
   890→            }
   891→            List<SpaceNodeEntity> spaceNodes = spaceNodeService.listByIds(relIds);
   892→            if (CollectionUtils.isEmpty(spaceNodes)) {
   893→                continue;
   894→            }
   895→            for (SpaceNodeEntity spaceNode : spaceNodes) {
   896→                if (StringUtils.isBlank(spaceNode.getName())) {
   897→                    continue;
   898→                }
   899→                occupiedNames.add(normalizeNameKey(spaceNode.getName()));
   900→            }
   901→        }
   902→        return occupiedNames;
   903→    }

   912→    private NameParts splitName(String name, Integer nodeType) {
   913→        if (SpaceNodeTypeEnum.FOLDER.getValue().equals(nodeType) || StringUtils.isBlank(name)) {
   914→            return splitConflictName(name, "");
   915→        }
   916→        int lastDotIndex = name.lastIndexOf('.');
   917→        if (lastDotIndex <= 0 || lastDotIndex >= name.length() - 1) {
   918→            return splitConflictName(name, "");
   919→        }
   920→        return splitConflictName(name.substring(0, lastDotIndex), name.substring(lastDotIndex));
   921→    }

   923→    private NameParts splitConflictName(String baseName, String suffix) {
   924→        if (StringUtils.isBlank(baseName)) {
   925→            return new NameParts(baseName, suffix);
   926→        }
   927→        Matcher matcher = CONFLICT_SUFFIX_PATTERN.matcher(baseName);
   928→        if (!matcher.matches()) {
   929→            return new NameParts(baseName, suffix);
   930→        }
   931→        String normalizedBaseName = matcher.group(1);
   932→        if (StringUtils.isBlank(normalizedBaseName)) {
   933→            return new NameParts(baseName, suffix);
   934→        }
   935→        return new NameParts(normalizedBaseName, suffix);
   936→    }

  1206→    private NextcloudPathInfo parsePathInfo(String fieldPath) {
  1207→        if (StringUtils.isBlank(fieldPath)) {
  1208→            log.warn("路径为空，无法查找space");
  1209→            return null;
  1210→        }
  1211→
  1212→        String[] pathParts = fieldPath.split("/");
  1213→        if (pathParts.length < MIN_PATH_PARTS) {
  1214→            log.warn("路径格式不正确，无法解析. path: {}", fieldPath);
  1215→            return null;
  1216→        }
  1217→
  1218→        int spaceNameIndex = -1;
  1219→        for (int i = 0; i < pathParts.length; i++) {
  1220→            if (NEXTCLOUD_FILES_MARKER.equals(pathParts[i]) && i + 1 < pathParts.length) {
  1221→                spaceNameIndex = i + 1;
  1222→                break;
  1223→            }
  1224→        }
  1225→
  1226→        if (spaceNameIndex == -1 || spaceNameIndex >= pathParts.length) {
  1227→            log.warn("无法从路径中提取spaceName. path: {}", fieldPath);
  1228→            return null;
  1229→        }
  1230→
  1231→        String spaceName = pathParts[spaceNameIndex];
  1232→        if (StringUtils.isBlank(spaceName)) {
  1233→            log.warn("spaceName为空. path: {}", fieldPath);
  1234→            return null;
  1235→        }
  1236→
  1237→        List<SpaceEntity> spaces = spaceService.matchByName(spaceName);
  1238→        if (CollectionUtils.isEmpty(spaces)) {
  1239→            log.warn("未找到对应的space. spaceName: {}", spaceName);
  1240→            return null;
  1241→        }
  1242→
  1243→        SpaceEntity spaceEntity = spaces.getFirst();
  1244→        String fileName = pathParts[pathParts.length - 1];
  1245→
  1246→        return new NextcloudPathInfo(spaceEntity, pathParts, spaceNameIndex, spaceName, fileName);
  1247→    }

  1264→    private boolean isDirectoryInNextcloud(String relativePath, String fileId, String spaceName) {
  1265→        NextcloudPropfindMultiStatus status = nextcloudService.propfind(relativePath, null, 0);
  1266→        if (status == null || CollectionUtils.isEmpty(status.getResponses())) {
  1267→            // 路径 propfind 失败（可能已被改名），尝试用 fileId 查当前路径后重试
  1268→            if (StringUtils.isNotBlank(fileId)) {
  1269→                String currentPath = nextcloudService.getFilePathByFileId(fileId, spaceName);
  1270→                if (StringUtils.isNotBlank(currentPath)) {
  1271→                    log.info("isDirectoryInNextcloud: 路径propfind未命中，通过fileId查到当前路径. fileId={}, originalPath={}, currentPath={}",
  1272→                            fileId, relativePath, currentPath);
  1273→                    status = nextcloudService.propfind(currentPath, null, 0);
  1274→                }
  1275→            }
  1276→            if (status == null || CollectionUtils.isEmpty(status.getResponses())) {
  1277→                log.info("isDirectoryInNextcloud: propfind最终未命中，默认判定为目录. fileId={}, relativePath={}", fileId, relativePath);
  1278→                return true;
  1279→            }
  1280→        }
  1281→        NextcloudPropfindMultiStatus.ResponseItem responseItem = status.getResponses().getFirst();
  1282→        if (responseItem == null || responseItem.getProp() == null) {
  1283→            log.info("isDirectoryInNextcloud: propfind响应缺少prop，默认判定为目录. fileId={}, relativePath={}", fileId, relativePath);
  1284→            return true;
  1285→        }
  1286→        boolean isDirectory = Boolean.TRUE.equals(responseItem.getProp().getDirectory());
  1287→        log.info("isDirectoryInNextcloud: 判定完成. fileId={}, relativePath={}, isDirectory={}", fileId, relativePath, isDirectory);
  1288→        return isDirectory;
  1289→    }
