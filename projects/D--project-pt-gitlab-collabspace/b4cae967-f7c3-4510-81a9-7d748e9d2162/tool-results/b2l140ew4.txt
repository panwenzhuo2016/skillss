+import org.springframework.amqp.core.Message;
+import org.springframework.amqp.rabbit.annotation.RabbitListener;
+import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
+import org.springframework.stereotype.Component;
+
+import java.io.IOException;
+
+import static com.yottastudios.ns.collabspace.shared.config.QueueConfig.DOC_SUBSCRIPTION_NOTIFY_QUEUE;
+
+@Slf4j
+@Component
+@ConditionalOnProperty(name = "mq.enabled", havingValue = "true", matchIfMissing = true)
+public class DocSubscriptionNotifyConsumer {
+
+    @Resource
+    private DocSubscriptionNotifyManager docSubscriptionNotifyManager;
+
+    @RabbitListener(queues = DOC_SUBSCRIPTION_NOTIFY_QUEUE)
+    public void handle(Message message, Channel channel) throws IOException {
+        long deliveryTag = message.getMessageProperties().getDeliveryTag();
+        String body = new String(message.getBody());
+        log.info("[订阅通知] 收到 MQ 消息: {}", body);
+        try {
+            DocSubscriptionNotifyMsg msg = JSONUtil.toBean(body, DocSubscriptionNotifyMsg.class);
+            docSubscriptionNotifyManager.handleDocUpdate(msg);
+            channel.basicAck(deliveryTag, false);
+        } catch (Exception e) {
+            log.error("[订阅通知] 消费 MQ 消息失败: {}", body, e);
+            channel.basicNack(deliveryTag, false, true);
+        }
+    }

+        List<DocSubscriptionEntity> subscribers = docSubscriptionService.listBySpaceNodeId(spaceNodeId);

+            return;

+            for (UserSimpleInfoVo u : allUsers) {

+        } catch (Exception e) {

+     * 发布文档更新通知消息。
+     * 若当前不在事务中，则直接发送。
+     *
+     * @param spaceNode  发生变更的文档节点
+     * @param operatorId 操作人 ID（通知时排除该用户），可为 null
+     */

+        if (spaceNode == null) {
+            return;

+                .spaceNodeId(spaceNode.getId())
+                .spaceNodeName(spaceNode.getName())
+                .spaceId(spaceNode.getSpaceId())
+                .spaceNodeType(spaceNode.getType())
+                .operatorId(operatorId)
+                .traceId(traceId)
+                .build();
+
+        String msgJson = JSONUtil.toJsonStr(msg);
+
+        if (TransactionSynchronizationManager.isActualTransactionActive()) {
+            TransactionSynchronizationManager.registerSynchronization(new TransactionSynchronization() {
+                @Override
+                public void afterCommit() {
+                    doSend(msgJson, spaceNode.getId());
