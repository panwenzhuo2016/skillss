    49→function parseCookies(res) {
    50→  const cookies = {};
    51→  const setCookieHeaders = res.headers.getSetCookie?.() || [];
    52→  for (const header of setCookieHeaders) {
    53→    const [kv] = header.split(';');
    54→    const eq = kv.indexOf('=');
    55→    if (eq > 0) {
    56→      cookies[kv.substring(0, eq).trim()] = kv.substring(eq + 1).trim();
    57→    }
    58→  }
    59→  return cookies;
    60→}

    62→async function rawFetch(method, url, body) {
    63→  const headers = {
    64→    'x-space-id': CONFIG.spaceId,
    65→    'content-type': 'application/json',
    66→  };
    67→  if (cookie) headers['cookie'] = cookie;
    68→  if (xsrfToken) headers['x-xsrf-token'] = xsrfToken;
    69→
    70→  const opts = { method, headers, redirect: 'manual' };
    71→  if (body) opts.body = JSON.stringify(body);
    72→  return fetch(url, opts);
    73→}

    75→// 安全 JSON 解析：将 16 位以上的裸数字转为字符串，避免 JS Long 精度丢失
    76→function safeJsonParse(text) {
    77→  return JSON.parse(text.replace(/:(\s*)(\d{16,})(\s*[,}\]])/g, ':$1"$2"$3'));
    78→}

    93→async function loginWithEmail(email) {
    94→  const res = await fetch(`${CONFIG.host}/api/v1/testLogin/${encodeURIComponent(email)}`,
    95→    { redirect: 'manual' });
    96→  const cookies = parseCookies(res);
    97→  if (!cookies.SESSION) return false;
    98→  xsrfToken = cookies['XSRF-TOKEN'] || '';
    99→  const tjwt = cookies['TJWT'] || '';
   100→  cookie = `SESSION=${cookies.SESSION}; XSRF-TOKEN=${xsrfToken}` + (tjwt ? `; TJWT=${tjwt}` : '');
   101→  return true;
   102→}

   130→function futureTime(minutesFromNow) {
   131→  const d = new Date(Date.now() + minutesFromNow * 60 * 1000);
   132→  const pad = n => String(n).padStart(2, '0');
   133→  return `${d.getFullYear()}-${pad(d.getMonth() + 1)}-${pad(d.getDate())} ${pad(d.getHours())}:${pad(d.getMinutes())}`;
   134→}

  1016→async function test_生命周期() {

  1022→  }));
  1023→  assert('lifecycle 创建成功', createRes.success === true, createRes);
  1024→  const id = createRes.success ? String(createRes.data?.id) : null;
  1025→  if (!id) return;
  1026→

  1028→  const active = await GET(`/space-announcement/active?category=1&spaceId=${CONFIG.spaceId}`);
  1029→  assert('lifecycle ACTIVE 状态 active 有数据', active.data != null && String(active.data.id) === id);
  1030→  const banner = await GET(`/space-announcement/banner?spaceId=${CONFIG.spaceId}`);
  1031→  assert('lifecycle ACTIVE 在 banner', banner.data?.some(a => String(a.id) === id));
  1032→  const unread = await GET(`/space-announcement/unread-popup?spaceId=${CONFIG.spaceId}`);
  1033→  assert('lifecycle ACTIVE+popup 在 unread', unread.data?.some(a => String(a.id) === id));

  1037→    id: id, title: '生命周期-已编辑', content: '<p>编辑后</p>', isPopup: false,
  1038→  });
  1039→  const detail = await GET(`/space-announcement/detail?id=${id}`);
  1040→  assert('lifecycle 编辑后标题正确', detail.data?.title === '生命周期-已编辑');
  1041→  assert('lifecycle 编辑后 isPopup=false', detail.data?.isPopup === false);
  1042→
  1043→  // 4. 下线
  1044→  await offlineById(id);
  1045→  const afterOff = await GET(`/space-announcement/detail?id=${id}`);
  1046→  assert('lifecycle 下线后 status=2', afterOff.data?.status === 2);
  1047→  const activeAfter = await GET(`/space-announcement/active?category=1&spaceId=${CONFIG.spaceId}`);
  1048→  assert('lifecycle 下线后 active=null', activeAfter.data == null);

  1057→}
