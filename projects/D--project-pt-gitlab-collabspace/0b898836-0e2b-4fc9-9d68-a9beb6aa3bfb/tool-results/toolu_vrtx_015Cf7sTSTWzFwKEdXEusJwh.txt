   794→+                    matchedRoleCodes.add(entry.getKey());
   795→+                }
   796→+            }
   797→+            if (keyword.contains("移除") || "移除所有权限".contains(keyword)) {
   798→+                // action=remove 表示移除权限，old_role/new_role 里不会存中文
   799→+                // 通过 action 字段匹配
   800→+                matchedRoleCodes.add("__REMOVE__");
   801→+            }
   802→+
   803→+            wrapper.and(w -> {
   804→+                w.like(NodePermissionLogEntity::getOperatorName, keyword)
   805→+                        .or().like(NodePermissionLogEntity::getTargetName, keyword)
   806→+                        .or().like(NodePermissionLogEntity::getNewRole, keyword)
   807→+                        .or().like(NodePermissionLogEntity::getOldRole, keyword);
   808→+                for (String roleCode : matchedRoleCodes) {
   809→+                    if ("__REMOVE__".equals(roleCode)) {
   810→+                        w.or().eq(NodePermissionLogEntity::getAction, "remove");
   811→+                    } else {
   812→+                        w.or().eq(NodePermissionLogEntity::getNewRole, roleCode)
   813→+                                .or().eq(NodePermissionLogEntity::getOldRole, roleCode);
   814→+                    }
   815→+                }
   816→+            });
   817→+        }
   818→+        wrapper.orderByDesc(NodePermissionLogEntity::getCreatedAt);
   819→+
   820→+        IPage<NodePermissionLogEntity> page = page(new Page<>(pageNo, pageSize), wrapper);
   821→+        IPage<NodePermissionLogVo> voPage = page.convert(this::toVo);
   822→+        return PageHelper.build(voPage);
   823→+    }

   834→+                .targetType(targetType)
   835→+                .action(action)
   836→+                .oldRole(oldRole)
   837→+                .newRole(newRole)
   838→+                .build();
   839→+        save(entity);
   840→+    }

   842→+    private NodePermissionLogVo toVo(NodePermissionLogEntity entity) {
   843→+        NodePermissionLogVo vo = new NodePermissionLogVo();

   847→+
   848→+        String action = entity.getAction();

   851→+            vo.setNewRoleDisplay("移除所有权限");

   855→+            vo.setNewRoleDisplay(roleDisplay);
   856→+        }
   857→+        return vo;
   858→+    }
