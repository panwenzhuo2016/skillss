   355→   355→    private List<CreatedIssueInfo> batchCreateJiraIssues(HttpServletRequest request) throws Exception {
   356→   356→        UserProfile userProfile = userManager.getRemoteUser(request);
   357→   357→        if (userProfile == null) throw new RuntimeException("未登录");
   358→   358→        ApplicationUser applicationUser = ComponentAccessor.getUserManager().getUserByName(userProfile.getUsername());
   359→   359→        if (applicationUser == null) throw new RuntimeException("无法获取应用用户");
   360→   360→        JiraAuthenticationContext jiraAuthenticationContext = ComponentAccessor.getJiraAuthenticationContext();
   361→   361→        jiraAuthenticationContext.setLoggedInUser(applicationUser);
   362→   362→
   363→   363→        String body = readBody(request);
   364→   364→        if (body == null || body.trim().isEmpty()) throw new RuntimeException("请求体不能为空");
   365→   365→        JSONObject root = new JSONObject(body);
   366→   366→        if (!root.has("issues")) throw new RuntimeException("issues不能为空");
   367→   367→        JSONArray arr = root.getJSONArray("issues");
   368→   368→        if (arr.length() == 0) throw new RuntimeException("issues不能为空");
   369→   369→
   370→   370→        List<CreateJiraIssueV2Ro> issueRos = new ArrayList<>();
   371→   371→        for (int i = 0; i < arr.length(); i++) {
   372→   372→            Object item = arr.get(i);
   373→   373→            if (!(item instanceof JSONObject)) throw new RuntimeException("issues项必须为对象");
   374→   374→            issueRos.add(CreateJiraIssueV2Ro.parseCreateJiraIssueV2((JSONObject) item));
   375→   375→        }
   376→   376→
   377→   377→        for (CreateJiraIssueV2Ro item : issueRos) {
   378→   378→            if (item.getIssueTemplateId() == null) throw new RuntimeException("每个issue必须包含issueTemplateId");
   379→   379→            if (item.getFields() == null) throw new RuntimeException("请求体字段不能为空");
   380→   380→        }
   381→   381→
   382→   382→        Map<Long, CreateJiraIssueV2Ro> templateIdToItem = new LinkedHashMap<>();
   383→   383→        for (CreateJiraIssueV2Ro item : issueRos) {
   384→   384→            Long id = item.getIssueTemplateId();
   385→   385→            if (templateIdToItem.containsKey(id)) throw new RuntimeException("issueTemplateId重复: " + id);
   386→   386→            templateIdToItem.put(id, item);
   387→   387→        }
   388→   388→
   389→   389→        Map<Long, List<Long>> edges = new HashMap<>();
   390→   390→        Map<Long, Integer> indegree = new HashMap<>();
   391→   391→        for (Long id : templateIdToItem.keySet()) indegree.put(id, 0);
   392→   392→        for (CreateJiraIssueV2Ro item : issueRos) {
   393→   393→            Long parent = item.getParentIssueTemplateId();
   394→   394→            if (parent != null) {
   395→   395→                if (!templateIdToItem.containsKey(parent)) throw new RuntimeException("未找到依赖的issueTemplateId: " + parent);
   396→   396→                edges.computeIfAbsent(parent, k -> new ArrayList<>()).add(item.getIssueTemplateId());
   397→   397→                indegree.put(item.getIssueTemplateId(), indegree.get(item.getIssueTemplateId()) + 1);
   398→   398→            }
   399→   399→            List<Long> ups = item.getUpstreamIssueTemplateIds();
   400→   400→            if (ups != null) {
   401→   401→                for (Long up : ups) {
   402→   402→                    if (up == null) continue;
   403→   403→                    if (!templateIdToItem.containsKey(up)) throw new RuntimeException("未找到依赖的issueTemplateId: " + up);
   404→   404→                    edges.computeIfAbsent(up, k -> new ArrayList<>()).add(item.getIssueTemplateId());
   405→   405→                    indegree.put(item.getIssueTemplateId(), indegree.get(item.getIssueTemplateId()) + 1);
   406→   406→                }
   407→   407→            }
   408→   408→        }
   409→   409→
   410→   410→        Set<Long> nodesWithChildren = new HashSet<>(edges.keySet());
   411→   411→        Set<Long> nodesReferencedAsChild = new HashSet<>();
   412→   412→        for (List<Long> children : edges.values()) nodesReferencedAsChild.addAll(children);
   413→   413→        Set<Long> inGraph = new HashSet<>();
   414→   414→        inGraph.addAll(nodesWithChildren);
   415→   415→        inGraph.addAll(nodesReferencedAsChild);
   416→   416→
   417→   417→        Deque<Long> queue = new ArrayDeque<>();

   827→   827→    private JSONObject deepCopyJson(JSONObject original) {
   828→   828→        log.debug("开始深拷贝JSON对象，原对象键数量: {}", original.keySet().size());
   829→   829→
   830→   830→        JSONObject copy = new JSONObject();
   831→   831→
   832→   832→        // 遍历原对象的所有键值对
   833→   833→        for (String key : original.keySet()) {
   834→   834→            Object value = original.get(key);
   835→   835→            log.debug("拷贝字段 '{}': {} (类型: {})", key, value, value != null ? value.getClass().getSimpleName() : "null");
   836→   836→
   837→   837→            if (value instanceof JSONObject) {
   838→   838→                log.debug("字段'{}'是JSONObject，进行递归拷贝", key);
   839→   839→                copy.put(key, deepCopyJson((JSONObject) value));
   840→   840→            } else if (value instanceof JSONArray) {
   841→   841→                log.debug("字段'{}'是JSONArray，进行数组拷贝", key);
   842→   842→                JSONArray originalArray = (JSONArray) value;
   843→   843→                JSONArray copyArray = new JSONArray();
   844→   844→                for (int i = 0; i < originalArray.length(); i++) {
   845→   845→                    Object arrayItem = originalArray.get(i);
   846→   846→                    if (arrayItem instanceof JSONObject) {
   847→   847→                        copyArray.put(deepCopyJson((JSONObject) arrayItem));
   848→   848→                    } else {
   849→   849→                        copyArray.put(arrayItem);
   850→   850→                    }
   851→   851→                }
   852→   852→                copy.put(key, copyArray);
   853→   853→            } else {
   854→   854→                log.debug("字段'{}'是基本类型，直接拷贝", key);
   855→   855→                copy.put(key, value);
   856→   856→            }
   857→   857→        }
   858→   858→
   859→   859→        log.debug("深拷贝完成，新对象键数量: {}", copy.keySet().size());
   860→   860→        return copy;
   861→   861→    }

  1003→  1003→    private String renderDependencyForest(Map<Long, List<Long>> edges, Set<Long> inGraph, List<Long> standalone) {
  1004→  1004→        StringBuilder sb = new StringBuilder();
  1005→  1005→        Set<Long> referenced = new HashSet<>();
  1006→  1006→        for (List<Long> children : edges.values()) referenced.addAll(children);
  1007→  1007→        List<Long> roots = new ArrayList<>();
  1008→  1008→        for (Long id : inGraph) if (!referenced.contains(id)) roots.add(id);
  1009→  1009→
  1010→  1010→        if (!roots.isEmpty()) sb.append("依赖树:\n");
  1011→  1011→        for (Long root : roots) appendDependencyTree(root, edges, "", sb);
  1012→  1012→
  1013→  1013→        if (!standalone.isEmpty()) {
  1014→  1014→            if (sb.length() > 0) sb.append('\n');
  1015→  1015→            sb.append("独立节点(无上下游): ");
  1016→  1016→            for (int i = 0; i < standalone.size(); i++) {
  1017→  1017→                if (i > 0) sb.append(", ");
  1018→  1018→                sb.append(standalone.get(i));
  1019→  1019→            }
  1020→  1020→        }
  1021→  1021→        return sb.toString();
  1022→  1022→    }

  1025→  1025→        sb.append(indent).append("- ").append(nodeId).append('\n');
  1026→  1026→        List<Long> children = edges.get(nodeId);

  1028→  1028→        String childIndent = indent + "  ";
  1029→  1029→        for (Long child : children) appendDependencyTree(child, edges, childIndent, sb);
