   191→    private boolean handleRequest(HttpServletRequest request, HttpServletResponse response,
   192→                                  Map<String, BiConsumer<HttpServletRequest, HttpServletResponse>> routes) {
   193→        String pathInfo = request.getPathInfo();
   194→        if (pathInfo == null) {
   195→            pathInfo = "/";
   196→        }
   197→
   198→        BiConsumer<HttpServletRequest, HttpServletResponse> handler = routes.get(pathInfo);
   199→        if (handler != null) {
   200→            handler.accept(request, response);
   201→            return true;
   202→        }
   203→
   204→        return false;
   205→    }

   355→    private List<CreatedIssueInfo> batchCreateJiraIssues(HttpServletRequest request) throws Exception {
   356→        UserProfile userProfile = userManager.getRemoteUser(request);
   357→        if (userProfile == null) throw new RuntimeException("未登录");
   358→        ApplicationUser applicationUser = ComponentAccessor.getUserManager().getUserByName(userProfile.getUsername());
   359→        if (applicationUser == null) throw new RuntimeException("无法获取应用用户");
   360→        JiraAuthenticationContext jiraAuthenticationContext = ComponentAccessor.getJiraAuthenticationContext();
   361→        jiraAuthenticationContext.setLoggedInUser(applicationUser);
   362→
   363→        String body = readBody(request);
   364→        if (body == null || body.trim().isEmpty()) throw new RuntimeException("请求体不能为空");
   365→        JSONObject root = new JSONObject(body);
   366→        if (!root.has("issues")) throw new RuntimeException("issues不能为空");
   367→        JSONArray arr = root.getJSONArray("issues");
   368→        if (arr.length() == 0) throw new RuntimeException("issues不能为空");
   369→
   370→        List<CreateJiraIssueV2Ro> issueRos = new ArrayList<>();
   371→        for (int i = 0; i < arr.length(); i++) {
   372→            Object item = arr.get(i);
   373→            if (!(item instanceof JSONObject)) throw new RuntimeException("issues项必须为对象");
   374→            issueRos.add(CreateJiraIssueV2Ro.parseCreateJiraIssueV2((JSONObject) item));
   375→        }
   376→
   377→        for (CreateJiraIssueV2Ro item : issueRos) {
   378→            if (item.getIssueTemplateId() == null) throw new RuntimeException("每个issue必须包含issueTemplateId");
   379→            if (item.getFields() == null) throw new RuntimeException("请求体字段不能为空");
   380→        }
   381→
   382→        Map<Long, CreateJiraIssueV2Ro> templateIdToItem = new LinkedHashMap<>();
   383→        for (CreateJiraIssueV2Ro item : issueRos) {
   384→            Long id = item.getIssueTemplateId();
   385→            if (templateIdToItem.containsKey(id)) throw new RuntimeException("issueTemplateId重复: " + id);
   386→            templateIdToItem.put(id, item);
   387→        }
   388→
   389→        Map<Long, List<Long>> edges = new HashMap<>();
   390→        Map<Long, Integer> indegree = new HashMap<>();
   391→        for (Long id : templateIdToItem.keySet()) indegree.put(id, 0);
   392→        for (CreateJiraIssueV2Ro item : issueRos) {
   393→            Long parent = item.getParentIssueTemplateId();
   394→            if (parent != null) {
   395→                if (!templateIdToItem.containsKey(parent)) throw new RuntimeException("未找到依赖的issueTemplateId: " + parent);
   396→                edges.computeIfAbsent(parent, k -> new ArrayList<>()).add(item.getIssueTemplateId());
   397→                indegree.put(item.getIssueTemplateId(), indegree.get(item.getIssueTemplateId()) + 1);
   398→            }
   399→            List<Long> ups = item.getUpstreamIssueTemplateIds();
   400→            if (ups != null) {
   401→                for (Long up : ups) {
   402→                    if (up == null) continue;
   403→                    if (!templateIdToItem.containsKey(up)) throw new RuntimeException("未找到依赖的issueTemplateId: " + up);
   404→                    edges.computeIfAbsent(up, k -> new ArrayList<>()).add(item.getIssueTemplateId());
   405→                    indegree.put(item.getIssueTemplateId(), indegree.get(item.getIssueTemplateId()) + 1);
   406→                }
   407→            }
   408→        }
   409→
   410→        Set<Long> nodesWithChildren = new HashSet<>(edges.keySet());
   411→        Set<Long> nodesReferencedAsChild = new HashSet<>();
   412→        for (List<Long> children : edges.values()) nodesReferencedAsChild.addAll(children);
   413→        Set<Long> inGraph = new HashSet<>();
   414→        inGraph.addAll(nodesWithChildren);
   415→        inGraph.addAll(nodesReferencedAsChild);
   416→
   417→        Deque<Long> queue = new ArrayDeque<>();
   418→        for (Long id : inGraph) if (indegree.get(id) == 0) queue.addLast(id);
   419→        List<Long> topoOrder = new ArrayList<>();
   420→        while (!queue.isEmpty()) {
   421→            Long u = queue.removeFirst();
   422→            topoOrder.add(u);
   423→            List<Long> children = edges.get(u);
   424→            if (children == null || children.isEmpty()) continue;
   425→            for (Long v : children) {
   426→                int d = indegree.get(v) - 1;
   427→                indegree.put(v, d);
   428→                if (d == 0) queue.addLast(v);
   429→            }
   430→        }
   431→        if (topoOrder.size() != inGraph.size()) throw new RuntimeException("存在循环依赖，请检查上游/父任务设置");
   432→
   433→        List<Long> standalone = new ArrayList<>();
   434→        for (CreateJiraIssueV2Ro item : issueRos) {
   435→            Long id = item.getIssueTemplateId();
   436→            boolean noParent = item.getParentIssueTemplateId() == null;
   437→            List<Long> ups = item.getUpstreamIssueTemplateIds();
   438→            boolean noUpstreams = ups == null || ups.isEmpty();
   439→            if (noParent && noUpstreams && !inGraph.contains(id)) standalone.add(id);
   440→        }
   441→        List<Long> orderedIds = new ArrayList<>();
   442→        orderedIds.addAll(topoOrder);
   443→        orderedIds.addAll(standalone);
   444→
   445→        String dependencyTreeLog = renderDependencyForest(edges, inGraph, standalone);
   446→        log.error("批量创建依赖关系树:\n{}", dependencyTreeLog);
   447→
   448→        for (Long id : orderedIds) {
   449→            CreateJiraIssueV2Ro item = templateIdToItem.get(id);
   450→            validateIssueFields(applicationUser, item.getFields());
   451→        }
   452→
   453→        Map<Long, String> templateIdToIssueKey = new HashMap<>();
   454→        List<CreatedIssueInfo> results = new ArrayList<>();
   455→        for (Long id : orderedIds) {
   456→            CreateJiraIssueV2Ro item = templateIdToItem.get(id);
   457→            Map<String, Object> fields = item.getFields();
   458→
   459→            Issue created = createIssueWithParameters(applicationUser, fields);
   460→
   461→            Long parent = item.getParentIssueTemplateId();
   462→            if (parent != null) {
   463→                String parentKey = templateIdToIssueKey.get(parent);
   464→                if (parentKey == null) throw new RuntimeException("内部错误：未找到依赖创建结果: " + parent);
   465→                createIssueLink(created.getKey(), parentKey, LINK_TYPE_PARENT, applicationUser);
   466→            }
   467→
   468→            List<Long> ups = item.getUpstreamIssueTemplateIds();
   469→            if (ups != null) {
   470→                for (Long up : ups) {
   471→                    if (up == null) continue;
   472→                    String upKey = templateIdToIssueKey.get(up);
   473→                    if (upKey == null) throw new RuntimeException("内部错误：未找到依赖创建结果: " + up);
   474→                    createIssueLink(created.getKey(), upKey, LINK_TYPE_UPSTREAM, applicationUser);
   475→                }
   476→            }
   477→
   478→            if (item.getLinks() != null) {
   479→                Set<String> autoPairs = new HashSet<>();
   480→                if (parent != null) autoPairs.add(LINK_TYPE_PARENT + "@" + templateIdToIssueKey.get(parent));
   481→                if (ups != null) {
   482→                    for (Long up : ups) {
   483→                        if (up == null) continue;
   484→                        String upKey = templateIdToIssueKey.get(up);
   485→                        if (upKey != null) autoPairs.add(LINK_TYPE_UPSTREAM + "@" + upKey);
   486→                    }
   487→                }
   488→                for (LinkCreateRo link : item.getLinks()) {
   489→                    if (link == null) continue;
   490→                    String type = link.getLinkTypeName();

   744→    /**
   745→     * 根据项目键处理字段值
   746→     * 从数组中找到以项目键开头的值，去掉前缀返回

   748→    private String processFieldForProjectKey(Object fieldValue, String projectKey) {
   749→        log.warn("      ~~~> 开始处理字段值匹配，项目键: '{}'", projectKey);
   750→        log.warn("      ~~~> 输入字段值: {} (类型: {})", fieldValue, fieldValue != null ? fieldValue.getClass().getSimpleName() : "null");
   751→
   752→        if (fieldValue == null) {
   753→            log.warn("      ~~~> 字段值为null，返回空字符串");
   754→            return "";
   755→        }
   756→
   757→        try {
   758→            List<String> valueList = new ArrayList<>();
   759→
   760→            // 解析字段值为字符串数组
   761→            if (fieldValue instanceof JSONArray) {
   762→                log.warn("      ~~~> 字段值是JSONArray类型");
   763→                JSONArray array = (JSONArray) fieldValue;
   764→                for (int i = 0; i < array.length(); i++) {
   765→                    Object item = array.get(i);
   766→                    if (item != null) {
   767→                        valueList.add(item.toString());
   768→                        log.warn("      ~~~> 从JSONArray中提取值[{}]: '{}'", i, item.toString());
   769→                    }
   770→                }
   771→            } else if (fieldValue instanceof List) {
   772→                log.warn("      ~~~> 字段值是List类型");
   773→                List<?> list = (List<?>) fieldValue;
   774→                for (int i = 0; i < list.size(); i++) {
   775→                    Object item = list.get(i);
   776→                    if (item != null) {
   777→                        valueList.add(item.toString());
   778→                        log.warn("      ~~~> 从List中提取值[{}]: '{}'", i, item.toString());
   779→                    }
   780→                }
   781→            } else if (fieldValue instanceof Object[]) {
   782→                log.warn("      ~~~> 字段值是Object[]类型");
   783→                Object[] array = (Object[]) fieldValue;
   784→                for (int i = 0; i < array.length; i++) {
   785→                    Object item = array[i];
   786→                    if (item != null) {
   787→                        valueList.add(item.toString());
   788→                        log.warn("      ~~~> 从Object[]中提取值[{}]: '{}'", i, item.toString());
   789→                    }
   790→                }
   791→            } else {
   792→                log.warn("      ~~~> 字段值是单个值类型");
   793→                valueList.add(fieldValue.toString());
   794→                log.warn("      ~~~> 单个值: '{}'", fieldValue.toString());
   795→            }
   796→
   797→            log.warn("      ~~~> 解析完成，值列表: {}", valueList);
   798→
   799→            // 在数组中查找以当前项目键开头的值
   800→            String prefix = projectKey + "-";
   801→            log.warn("      ~~~> 查找前缀: '{}'", prefix);
   802→
   803→            for (int i = 0; i < valueList.size(); i++) {
   804→                String value = valueList.get(i);
   805→                log.warn("      ~~~> 检查值[{}]: '{}' 是否以前缀'{}'开头", i, value, prefix);
   806→
   807→                if (value != null && value.startsWith(prefix)) {
   808→                    String result = value.substring(prefix.length());
   809→                    log.warn("      ~~~> ✓ 找到匹配值！原值: '{}' -> 去前缀后: '{}'", value, result);
   810→                    return result;
   811→                }
   812→            }
   813→
   814→            log.warn("      ~~~> ✗ 没找到以'{}'开头的匹配值，返回空字符串", prefix);
   815→            return "";
   816→
   817→        } catch (Exception e) {
   818→            log.error("      ~~~> 处理字段值时发生错误，字段值: {}, 项目键: {}, 错误: {}",
   819→                    fieldValue, projectKey, e.getMessage(), e);
   820→            return "";
   821→        }
   822→    }

   824→    /**
   825→     * JSON深拷贝方法，保留所有字段
   826→     */
   827→    private JSONObject deepCopyJson(JSONObject original) {
   828→        log.debug("开始深拷贝JSON对象，原对象键数量: {}", original.keySet().size());
   829→
   830→        JSONObject copy = new JSONObject();
   831→
   832→        // 遍历原对象的所有键值对
   833→        for (String key : original.keySet()) {
   834→            Object value = original.get(key);
   835→            log.debug("拷贝字段 '{}': {} (类型: {})", key, value, value != null ? value.getClass().getSimpleName() : "null");
   836→
   837→            if (value instanceof JSONObject) {
   838→                log.debug("字段'{}'是JSONObject，进行递归拷贝", key);
   839→                copy.put(key, deepCopyJson((JSONObject) value));
   840→            } else if (value instanceof JSONArray) {
   841→                log.debug("字段'{}'是JSONArray，进行数组拷贝", key);
   842→                JSONArray originalArray = (JSONArray) value;
   843→                JSONArray copyArray = new JSONArray();
   844→                for (int i = 0; i < originalArray.length(); i++) {
   845→                    Object arrayItem = originalArray.get(i);
   846→                    if (arrayItem instanceof JSONObject) {
   847→                        copyArray.put(deepCopyJson((JSONObject) arrayItem));
   848→                    } else {
   849→                        copyArray.put(arrayItem);
   850→                    }
   851→                }
   852→                copy.put(key, copyArray);
   853→            } else {
   854→                log.debug("字段'{}'是基本类型，直接拷贝", key);
   855→                copy.put(key, value);
   856→            }
   857→        }
   858→
   859→        log.debug("深拷贝完成，新对象键数量: {}", copy.keySet().size());
   860→        return copy;
   861→    }

   953→        List<String> result = new ArrayList<>();
   954→
   955→        if (fieldValue == null) {
   956→            return result;

   974→                }
   975→            }
   976→        }
   977→        // 字符串类型
   978→        else {
   979→            result.add(fieldValue.toString());
   980→        }
   981→
   982→        return result;

  1003→    private String renderDependencyForest(Map<Long, List<Long>> edges, Set<Long> inGraph, List<Long> standalone) {
  1004→        StringBuilder sb = new StringBuilder();
  1005→        Set<Long> referenced = new HashSet<>();
  1006→        for (List<Long> children : edges.values()) referenced.addAll(children);
  1007→        List<Long> roots = new ArrayList<>();
  1008→        for (Long id : inGraph) if (!referenced.contains(id)) roots.add(id);
  1009→
  1010→        if (!roots.isEmpty()) sb.append("依赖树:\n");
  1011→        for (Long root : roots) appendDependencyTree(root, edges, "", sb);
  1012→
  1013→        if (!standalone.isEmpty()) {
  1014→            if (sb.length() > 0) sb.append('\n');
  1015→            sb.append("独立节点(无上下游): ");
  1016→            for (int i = 0; i < standalone.size(); i++) {
  1017→                if (i > 0) sb.append(", ");
  1018→                sb.append(standalone.get(i));
  1019→            }
  1020→        }
  1021→        return sb.toString();
  1022→    }

  1024→    private void appendDependencyTree(Long nodeId, Map<Long, List<Long>> edges, String indent, StringBuilder sb) {
  1025→        sb.append(indent).append("- ").append(nodeId).append('\n');
  1026→        List<Long> children = edges.get(nodeId);
  1027→        if (children == null || children.isEmpty()) return;
  1028→        String childIndent = indent + "  ";
  1029→        for (Long child : children) appendDependencyTree(child, edges, childIndent, sb);
  1030→    }

  1344→        for (Locale loc : locales) {

  1349→                    Date dt = parser.parse(s);
  1350→                    SimpleDateFormat formatter = new SimpleDateFormat("yyyy/MM/dd", Locale.CHINA);

  1353→                }
  1354→            }
  1355→        }
  1356→        // 如果无法解析，原样返回，让Jira给出校验错误
