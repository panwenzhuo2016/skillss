   292→    renderMd(text: string): string {

   299→    // gzip 压缩字符串，返回 "gz:<base64>"
   300→    async compressContext(str: string): Promise<string> {
   301→        if (!str) {return str}
   302→        // 先去除多余空白，减少压缩前体积
   303→        str = str.replaceAll(/\n{3,}/g, '\n\n').trim()
   304→        const stream = new (window as any).CompressionStream('gzip')
   305→        const writer = stream.writable.getWriter()
   306→        writer.write(new TextEncoder().encode(str))
   307→        writer.close()
   308→        const chunks: Uint8Array[] = []
   309→        const reader = stream.readable.getReader()
   310→        while (true) {
   311→            const { done, value } = await reader.read()
   312→            if (done) {break}
   313→            chunks.push(value)
   314→        }
   315→        const merged = new Uint8Array(chunks.reduce((n, c) => n + c.length, 0))
   316→        let offset = 0
   317→        for (const c of chunks) { merged.set(c, offset); offset += c.length }
   318→        let binary = ''
   319→        for (let i = 0; i < merged.length; i++) {binary += String.fromCharCode(merged[i])}
   320→        return 'gz:' + btoa(binary)
   321→    }

   506→    /** 检查累积上下文是否可能超限，返回 'ok' 继续发送 / 'newSession' 新建会话 / 'abort' 关闭弹窗不操作 */
   507→    private async checkContextOverLimit(extraChars = 0): Promise<'ok' | 'newSession' | 'abort'> {
   508→        const contextCharLimit = 2_200_000
   509→        if (this.accumulatedContextChars + extraChars <= contextCharLimit) {return 'ok'}
   510→        try {
   511→            await this.$confirm(

   527→    private async flushSlotMessages() {
   528→        if (this.pendingSlotMessages.length === 0 || !this.currentRole) {return}
   529→        const sessionId = await dailyReportAiApis.saveMessages({
   530→            role_id: this.currentRole.id,
   531→            session_id: this.currentSessionId,
   532→            messages: this.pendingSlotMessages,
   533→        })
   534→        if (sessionId && !this.currentSessionId) {
   535→            this.currentSessionId = sessionId
   536→            await this.loadSessions()
   537→        }
   538→        this.pendingSlotMessages = []

   541→    private async sendExtractSlots(text: string) {
   542→        const limitAction = await this.checkContextOverLimit()
   543→        if (limitAction === 'newSession') { this.newSession(); return }
   544→        if (limitAction === 'abort') { return }
   545→
   546→        this.messages.push({ role: 'user', content: text })
   547→        this.pendingSlotMessages.push({ role: 'user', content: text })
   548→        this.inputText = ''
   549→        this.$nextTick(this.scrollToBottom)
   550→
   551→        const assistantIdx = this.messages.length
   552→        this.messages.push({ role: 'assistant', content: '' })
   553→        this.isStreaming = true
   554→
   555→        try {
   556→            // 收集可用分组名和所有成员花名
   557→            const availableGroups = this.groupOptions.map(g => g.label)
   558→            const availableMembers = [...new Set(this.groupOptions.flatMap(g => g.value))].filter(Boolean)
   559→            // 续查模式：传入上次的槽位结果
   560→            const previousContext = this.lastSlotResult ? {
   561→                member_type: this.lastSlotResult.member_type,
   562→                members: this.lastSlotResult.members,
   563→                period_start: this.lastSlotResult.period_start,
   564→                period_end: this.lastSlotResult.period_end,
   565→            } : undefined
   566→            const hasExistingContext = this.messages.some(m => m.role === 'context')
   567→            const result = await dailyReportAiApis.extractSlots({
   568→                message: text,
   569→                conversation_id: this.slotsConversationId || undefined,
   570→                has_existing_context: hasExistingContext || undefined,
   571→                available_groups: availableGroups,
   572→                available_members: availableMembers,
   573→                previous_context: previousContext,
   574→            })
   575→
   576→            if (!result) {
   577→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   578→                this.pendingSlotMessages.push({ role: 'assistant', content: this.messages[assistantIdx].content })
   579→                await this.flushSlotMessages()
   580→                return
   581→            }
   582→
   583→            // 保存 conversation_id 用于多轮
   584→            this.slotsConversationId = result.conversation_id
   585→
   586→            // 意图为分析已有数据 → 直接走问答，不重新取数据
   587→            if (result.intent === 'analyze') {
   588→                // 移除占位的 assistant 消息，sendAskMessage 会重新创建
   589→                this.messages.splice(assistantIdx, 1)
   590→                this.isStreaming = false
   591→                await this.flushSlotMessages()
   592→                await this.sendAskMessage(text, true)
   593→                return
   594→            }
   595→
   596→            if (!result.slots_complete) {
   597→                // 槽位不完整，显示反问
   598→                this.messages[assistantIdx].content = result.reply
   599→                this.pendingSlotMessages.push({ role: 'assistant', content: result.reply })
   600→                await this.flushSlotMessages()
   601→                return
   602→            }
   603→
   604→            // 槽位完整，保存结果用于续查
   605→            this.lastSlotResult = result
   606→            // 回显意图识别结果
   607→            const nicknames = this.resolveMembers(result)
   608→            const memberLabel = result.member_type === 'group'
   609→                ? `分组：${result.members.join('、')}`
   610→                : `成员：${result.members.join('、')}`
   611→            const countLabel = nicknames.length >= 10 ? `，共 ${nicknames.length} 人` : ''
   612→            const periodLabel = `${result.period_start} ~ ${result.period_end}`
   613→            const slotSummary = `查询对象：${memberLabel}${countLabel}\n时间范围：${periodLabel}${result.period_raw ? `（${result.period_raw}）` : ''}`
   614→            this.messages[assistantIdx].content = slotSummary + '\n\n正在同步日报数据...'
   615→            this.pendingSlotMessages.push({ role: 'assistant', content: slotSummary })
   616→            this.$nextTick(this.scrollToBottom)
   617→
   618→            if (nicknames.length === 0) {
   619→                const errMsg = `未找到匹配的成员「${result.members.join('、')}」，请尝试使用花名或先手动同步数据。`
   620→                this.messages[assistantIdx].content = errMsg
   621→                this.pendingSlotMessages.push({ role: 'assistant', content: errMsg })
   622→                await this.flushSlotMessages()
   623→                return
   624→            }
   625→
   626→            const context = await this.fetchReportContext(nicknames, result.period_start, result.period_end)
   627→            if (!context) {
   628→                this.messages[assistantIdx].content = '该时间段内没有找到日报数据。'
   629→                this.pendingSlotMessages.push({ role: 'assistant', content: '该时间段内没有找到日报数据。' })
   630→                await this.flushSlotMessages()
   631→                return
   632→            }
   633→
   634→            // 重置意图识别状态
   635→            this.slotsConversationId = ''
   636→
   637→            // 意图识别消息先落库（会创建 session）
   638→            await this.flushSlotMessages()
   639→
   640→            // 保留意图识别结果消息，走正常 ask 流程
   641→            this.reportContext = context
   642→            // 意图已由 extract-slots 处理，ask 只需分析同步好的日报数据
   643→            await this.sendAskMessage('请根据以上日报数据进行分析')
   644→        } catch {
   645→            this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   646→            this.pendingSlotMessages.push({ role: 'assistant', content: this.messages[assistantIdx].content })
   647→            await this.flushSlotMessages()
   648→        } finally {
   649→            this.isStreaming = false
   650→        }
   651→    }

   653→    /**
   654→     * 根据槽位结果解析出 nicknames 列表
   655→     */
   656→    private resolveMembers(result: ExtractSlotsResult): string[] {
   657→        if (result.member_type === 'group') {
   658→            // 匹配 groupOptions 的 label，展开为花名列表
   659→            const matched = this.groupOptions.filter(g =>
   660→                result.members.some(m => g.label.includes(m) || m.includes(g.label))
   661→            )
   662→            return [...new Set(matched.flatMap(g => g.value))].filter(Boolean)
   663→        }
   664→        // individual：LLM 返回的成员名必须在花名列表中匹配到，匹配不到则丢弃
   665→        const allMembers = this.groupOptions.flatMap(g => g.value).filter(Boolean)
   666→        return result.members
   667→            .map(name => allMembers.find(m => m.includes(name) || name.includes(m)))
   668→            .filter((n): n is string => !!n)
   669→    }

   671→    /**
   672→     * 根据 nicknames + 日期范围拉取日报数据，返回聚合后的文本上下文
   673→     */
   674→    private async fetchReportContext(nicknames: string[], startDate: string, endDate: string): Promise<string> {
   675→        const days = moment(endDate).diff(moment(startDate), 'day') + 1
   676→        const perPage = Math.max(200, days + 10)
   677→
   678→        const results = await Promise.all(
   679→            nicknames.map(nickname =>
   680→                dailyReportApis.getOtherReportsAPI({
   681→                    start_date: startDate,
   682→                    end_date: endDate,
   683→                    nickname,
   684→                    per_page: perPage,
   685→                    page: 1,
   686→                    show_rest: 1,
   687→                    sys_site: this.sysSite,
   688→                } as any)
   689→            )
   690→        )
   691→
   692→        const dateMap: Record<string, string[]> = {}
   693→        results.forEach((res, idx) => {
   694→            if (!res) {return}
   695→            const name = nicknames[idx].replace(/\[.+]/, '')
   696→            ;(res.reports ?? []).forEach((item: any) => {
   697→                const date = item.date ?? ''
   698→                if (!dateMap[date]) {dateMap[date] = []}
   699→                if (item.report) {
   700→                    dateMap[date].push(`${name}：\n${item.report.content ?? ''}`)
   701→                } else if ([4, 5].includes(item.type_id)) {
   702→                    dateMap[date].push(`${name}：休息日`)
   703→                } else {
   704→                    dateMap[date].push(`${name}：未提交`)
   705→                }
   706→            })
   707→        })
   708→
   709→        const sortedDates = Object.keys(dateMap).sort((a, b) => b.localeCompare(a))
   710→        return sortedDates
   711→            .map(date => `【${date}】\n${dateMap[date].join('\n\n')}`)
   712→            .join('\n\n---\n\n')
   713→    }

   715→    /**
   716→     * 正常问答流程：带 reportContext 调 askAi（SSE 流式）
   717→     */
   718→    private async sendAskMessage(text: string, skipUserPush = false) {
   719→        const contextToSend = this.reportContext
   720→        // 记录本轮消息插入起点，取消时用于回退
   721→        const msgStartIdx = this.messages.length
   722→        if (contextToSend) {
   723→            const contextLimit = 5000
   724→            const isTruncated = contextToSend.length > contextLimit
   725→            this.messages.push({
   726→                role: 'context',
   727→                content: isTruncated ? contextToSend.slice(0, contextLimit) : contextToSend,
   728→                _truncated: isTruncated,
   729→                _fullLength: contextToSend.length,
   730→                _fullContent: isTruncated ? contextToSend : undefined,
   731→            })
   732→            this.reportContext = ''
   733→        }
   734→        if (!skipUserPush) {
   735→            this.messages.push({ role: 'user', content: text })
   736→            this.inputText = ''
   737→        }
   738→        this.$nextTick(this.scrollToBottom)
   739→
   740→        const assistantIdx = this.messages.length
   741→        this.messages.push({ role: 'assistant', content: '' })
   742→        this.isStreaming = true
   743→
   744→        const newContextChars = contextToSend ? contextToSend.length : 0
   745→        const limitAction = await this.checkContextOverLimit(newContextChars)
   746→        if (limitAction !== 'ok') {
   747→            this.messages.splice(msgStartIdx)
   748→            this.isStreaming = false
   749→            if (limitAction === 'newSession') { this.newSession() }
   750→            return
   751→        }
   752→
   753→        this.streamAbort = new AbortController()
   754→        try {
   755→            const history = this.messages.slice(0, -2).filter(m => m.role !== 'context' && !!m.content).slice(-10) as { role: 'user' | 'assistant'; content: string }[]
   756→            const processedContext = contextToSend ? compressDailyContext(contextToSend) : ''
   757→            if (processedContext) {
   758→                console.log('[AiChat] 压缩前上下文:\n', processedContext)
   759→            }
   760→            const compressedContext = processedContext ? await this.compressContext(processedContext) : ''
   761→            const response = await dailyReportAiApis.askAi({
   762→                role_id: this.currentRole!.id,
   763→                session_id: this.currentSessionId,
   764→                message: text,
   765→                report_context: compressedContext,
   766→                report_context_raw: contextToSend ? await this.compressContext(contextToSend) : undefined,
   767→                system_prompt: this.currentRole!.system_prompt,
   768→                history,
   769→            }, this.streamAbort.signal)
   770→
   771→            const reader = response.body!.getReader()
   772→            const decoder = new TextDecoder()
   773→            let buffer = ''
   774→
   775→            while (true) {
   776→                const { done, value } = await reader.read()
   777→                if (done) {break}
   778→                buffer += decoder.decode(value, { stream: true })
   779→                const lines = buffer.split('\n')
   780→                buffer = lines.pop() ?? ''
   781→
   782→                for (const line of lines) {
   783→                    if (!line.startsWith('data: ')) {continue}
   784→                    try {
   785→                        const parsed = JSON.parse(line.slice(6))
   786→                        if (parsed.error_code !== 0) {
   787→                            this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   788→                            break
   789→                        }
   790→                        const data = parsed.data
   791→                        // TransferRequestService SSE 格式：meta（首帧，含 session_id）+ chat（流式内容）
   792→                        if (data.type === 'meta') {
   793→                            if (data.session_id && !this.currentSessionId) {
   794→                                this.currentSessionId = Number(data.session_id)
   795→                                this.loadSessions()
   796→                            }
   797→                        } else if (data.type === 'chat' && data.content) {
   798→                            this.messages[assistantIdx].content += data.content
   799→                            this.$nextTick(this.scrollToBottom)
   800→                        }
   801→                    } catch {}
   802→                }
   803→            }
   804→            // 流式读取完成，累加本次上下文字符数
   805→            if (newContextChars > 0) {
   806→                this.accumulatedContextChars += newContextChars
   807→            }
   808→        } catch (e: any) {
   809→            if (e?.name !== 'AbortError') {
   810→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   811→            }
   812→        } finally {
   813→            if (!this.messages[assistantIdx]?.content && this.streamAbort && !this.streamAbort.signal.aborted) {
   814→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   815→            }
   816→            this.streamAbort = null
   817→            this.isStreaming = false
   818→            // AI 回复落库：DB 存储现在由前端完成
   819→            const finalContent = this.messages[assistantIdx]?.content
   820→            if (finalContent && this.currentSessionId) {
   821→                await dailyReportAiApis.saveMessages({
   822→                    role_id: this.currentRole!.id,
   823→                    session_id: this.currentSessionId,
   824→                    messages: [{ role: 'assistant', content: finalContent }],
   825→                })
   826→            }
   827→            await this.loadSessions()
   828→        }
   829→    }

   831→    // 加载更多 context 内容
   832→    async loadMoreContext(m: { content: string; _id?: number; _truncated?: boolean; _fullLength?: number; _fullContent?: string; _loading?: boolean }) {
   833→        if (!m._truncated || m._loading) {return}
   834→        const chunkSize = 5000
   835→
   836→        // 实时场景：完整内容在本地 _fullContent 中，直接截取
   837→        if (m._fullContent) {
   838→            const offset = m.content.length
   839→            const nextChunk = m._fullContent.slice(offset, offset + chunkSize)
   840→            m.content += nextChunk
   841→            m._truncated = (offset + chunkSize) < m._fullContent.length
   842→            if (!m._truncated) {m._fullContent = undefined} // 加载完毕，释放内存
   843→            return
   844→        }
   845→
   846→        // 历史场景：从后端分段加载
   847→        if (!m._id) {return}
   848→        this.$set(m, '_loading', true)
   849→        try {
   850→            const offset = m.content.length
   851→            const result = await dailyReportAiApis.getContextMore(m._id, offset)
   852→            if (result) {
   853→                m.content += result.content
   854→                m._truncated = result.has_more
   855→                m._fullLength = result.full_length
   856→            }
   857→        } finally {
   858→            this.$set(m, '_loading', false)
   859→        }
   860→    }
