    57→创建 `oa-core/app/Services/AiAnalysisService.php`：
    58→
    59→```php
    60→<?php
    61→
    62→namespace App\Services;
    63→
    64→use Illuminate\Support\Arr;
    65→use Illuminate\Support\Facades\Log;
    66→
    67→class AiAnalysisService
    68→{
    69→    use ServiceTrait;
    70→
    71→    /**
    72→     * Blocking 模式：调用 Vertex AI，返回完整分析结果。
    73→     * 用于邮件发送场景。
    74→     */
    75→    public function analyze(string $content, string $systemPrompt): string
    76→    {
    77→        $endpoint = env('VERTEX_AI_ENDPOINT');
    78→        $apiKey = env('VERTEX_AI_API_KEY');
    79→        $model = env('VERTEX_AI_MODEL', 'google/gemini-2.0-flash');
    80→
    81→        $payload = json_encode([
    82→            'model' => $model,
    83→            'messages' => [
    84→                ['role' => 'system', 'content' => $systemPrompt],
    85→                ['role' => 'user', 'content' => $content],
    86→            ],
    87→            'stream' => false,
    88→        ], JSON_UNESCAPED_UNICODE);
    89→
    90→        Log::info('[AiAnalysis] blocking request', [
    91→            'content_len' => mb_strlen($content),
    92→            'prompt_len' => mb_strlen($systemPrompt),
    93→        ]);
    94→
    95→        $ch = curl_init($endpoint);
    96→        curl_setopt_array($ch, [
    97→            CURLOPT_POST => true,
    98→            CURLOPT_POSTFIELDS => $payload,
    99→            CURLOPT_HTTPHEADER => [
   100→                'Content-Type: application/json',
   101→                'Authorization: Bearer ' . $apiKey,
   102→            ],
   103→            CURLOPT_RETURNTRANSFER => true,
   104→            CURLOPT_TIMEOUT => 120,
   105→        ]);
   106→
   107→        $response = curl_exec($ch);
   108→        $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
   109→        $curlError = $response === false ? curl_error($ch) : '';
   110→        curl_close($ch);
   111→
   112→        if ($curlError) {
   113→            Log::error('[AiAnalysis] curl error', ['error' => $curlError]);
   114→            return '';
   115→        }
   116→
   117→        $result = json_decode($response, true);
   118→        if (!is_array($result)) {
   119→            Log::error('[AiAnalysis] invalid response', ['raw' => mb_substr($response, 0, 1000)]);
   120→            return '';
   121→        }
   122→
   123→        $answer = Arr::get($result, 'choices.0.message.content', '');
   124→        Log::info('[AiAnalysis] blocking done', ['answer_len' => mb_strlen($answer)]);
   125→
   126→        return $answer;
   127→    }
   128→
   129→    /**
   130→     * Streaming 模式：调用 Vertex AI，通过回调逐 chunk 输出。
   131→     * 用于测试 Prompt 和前端 SSE 场景。
   132→     *
   133→     * @param string $content 日报内容
   134→     * @param string $systemPrompt AI Prompt
   135→     * @param callable $onChunk fn(string $chunk): void 每个 chunk 的回调
   136→     * @param array $history 可选的历史对话消息
   137→     */
   138→    public function analyzeStream(string $content, string $systemPrompt, callable $onChunk, array $history = []): string
   139→    {
   140→        $endpoint = env('VERTEX_AI_ENDPOINT');
   141→        $apiKey = env('VERTEX_AI_API_KEY');
   142→        $model = env('VERTEX_AI_MODEL', 'google/gemini-2.0-flash');
   143→
   144→        $messages = [
   145→            ['role' => 'system', 'content' => $systemPrompt],
   146→        ];
   147→        foreach ($history as $msg) {
   148→            $messages[] = ['role' => $msg['role'], 'content' => $msg['content']];
   149→        }
   150→        if (!empty($content)) {
   151→            $messages[] = ['role' => 'user', 'content' => $content];
   152→        }
   153→
   154→        $payload = json_encode([
   155→            'model' => $model,
   156→            'messages' => $messages,
   157→            'stream' => true,
   158→        ], JSON_UNESCAPED_UNICODE);
   159→
   160→        Log::info('[AiAnalysis] streaming request', [
   161→            'content_len' => mb_strlen($content),
   162→            'prompt_len' => mb_strlen($systemPrompt),
   163→            'history_count' => count($history),
   164→        ]);
   165→
   166→        $fullReply = '';
   167→        $buffer = '';
   168→
   169→        $ch = curl_init($endpoint);
   170→        curl_setopt_array($ch, [
   171→            CURLOPT_POST => true,
   172→            CURLOPT_POSTFIELDS => $payload,
   173→            CURLOPT_HTTPHEADER => [
   174→                'Content-Type: application/json',
   175→                'Authorization: Bearer ' . $apiKey,
   176→            ],
   177→            CURLOPT_RETURNTRANSFER => false,
   178→            CURLOPT_TIMEOUT => 300,
   179→            CURLOPT_WRITEFUNCTION => function ($curl, $data) use (&$fullReply, &$buffer, $onChunk) {
   180→                $buffer .= $data;
   181→
   182→                while (($pos = strpos($buffer, "\n")) !== false) {
   183→                    $line = rtrim(substr($buffer, 0, $pos), "\r");
   184→                    $buffer = substr($buffer, $pos + 1);
   185→
   186→                    if (!str_starts_with($line, 'data: ')) {
   187→                        continue;
   188→                    }
   189→
   190→                    $jsonStr = substr($line, 6);
   191→                    if ($jsonStr === '[DONE]') {
   192→                        continue;
   193→                    }
   194→
   195→                    $decoded = json_decode($jsonStr, true);
   196→                    if (!is_array($decoded)) {
   197→                        continue;
   198→                    }
   199→
   200→                    $chunk = Arr::get($decoded, 'choices.0.delta.content', '');
   201→                    if ($chunk !== '') {
   202→                        $fullReply .= $chunk;
   203→                        $onChunk($chunk);
   204→                    }
   205→                }
   206→
   207→                return strlen($data);
   208→            },
   209→        ]);
   210→
   211→        curl_exec($ch);
   212→        $curlError = curl_error($ch);
   213→        curl_close($ch);
   214→
   215→        if ($curlError) {
   216→            Log::error('[AiAnalysis] streaming curl error', ['error' => $curlError]);
   217→        }
   218→
   219→        Log::info('[AiAnalysis] streaming done', ['reply_len' => mb_strlen($fullReply)]);
   220→
   221→        return $fullReply;
   222→    }
   223→}
   224→```

   727→class DailyReportAiEmailJob extends MyJob
   728→{
   729→    public function handle(): bool
   730→    {
   731→        $now = now();
   732→        $currentTime = $now->format('H:i');
   733→        $currentDayOfWeek = $now->dayOfWeekIso; // 1=Monday, 7=Sunday
   734→        $currentDayOfMonth = $now->day;
   735→

   748→            // 检查发送周期是否匹配

   764→    private function shouldSendNow(string $sendCycle, int $dayOfWeek, int $dayOfMonth): bool
   765→    {
   766→        return match ($sendCycle) {
   767→            'daily' => true,
   768→            'weekly_monday' => $dayOfWeek === 1,
   769→            'weekly_friday' => $dayOfWeek === 5,
   770→            'monthly_first' => $dayOfMonth === 1,
   771→            default => false,
   772→        };

   775→    private function processTask(object $task): void
   776→    {
   777→        $sfuserId = $task->sfuser_id;
   778→
   779→        // 读取邮箱配置

   781→        if (!$mc || empty(Arr::get($mc->config, 'to_email'))) {
   782→            info("[DailyReportAiEmailJob] task={$task->id} no email config, skip");
   783→            return;
   784→        }
   785→
   786→        $toEmails = Arr::get($mc->config, 'to_email');
   787→        $sfuserToken = Arr::get($mc->config, 'sfuser_authorized_token', '');
   788→        if (!$sfuserToken) {
   789→            info("[DailyReportAiEmailJob] task={$task->id} no sfuser_token, skip");
   790→            return;
   791→        }
   792→
   793→        // 确定日报日期范围
   794→        [$startDate, $endDate] = $this->getDateRange($task->report_range);
   795→
   796→        // 拉取日报数据
   797→        $dailyService = app(DailyReportService::class);
   798→        $focusUserIds = $dailyService->getSubordinates($sfuserId);
   799→        $data = $dailyService->getOthersDailyReportListEnable(
   800→            $startDate,
   801→            $endDate,
   802→            $sfuserId,
   803→            $focusUserIds['focus_users'],
   804→            10000
   805→        );
   806→
   807→        if (empty($data) || empty($data['reports']) || empty($data['reports']['report_list'])) {
   808→            info("[DailyReportAiEmailJob] task={$task->id} no report data, skip");
   809→            return;
   810→        }
   811→
   812→        // 构建邮件内容（加密格式，由 oa-core 解密）
   813→        $content = '';
   814→        foreach ($data['reports']['report_list'] as $report) {
   815→            $nickname = $report['data']['nickname'] ?? '';
   816→            if (empty($nickname)) {
   817→                continue;
   818→            }
   819→            $nickname = EncryptService::app()->markEncrString($nickname);
   820→            $reportContent = EncryptService::app()->markEncrString($report['content']);
   821→            $content .= '<p><b>' . $nickname . '</b></p>';
   822→            $content .= '<div style="border:1px solid #eee;padding:8px;margin-bottom:12px;">' . $reportContent . '</div>';
   823→        }
   824→
   825→        $subject = "【{$startDate}】 {$task->name} - AI 分析";
   826→
   827→        // 调 oa-core sendDecr，带 ai_analysis_config
   828→        info("[DailyReportAiEmailJob] task={$task->id} sending email to " . implode(',', $toEmails));
   829→        sendEmail(
   830→            $content,
   831→            $subject,
   832→            [$sfuserToken],
   833→            filter: 'emails',
   834→            filterMap: ['emails' => $toEmails],
   835→            // ai_analysis_config 需要在 sendEmail 函数中支持透传
   836→            // 注意：这里需要确认 sendEmail/getEmailParams 是否支持额外参数
   837→            // 如果不支持，需要直接调用 CoreApiLib
   838→        );
   839→        info("[DailyReportAiEmailJob] task={$task->id} email sent");
   840→    }

   843→    {
   844→        $today = now()->toDateString();

   846→            'today' => [$today, $today],
   847→            'yesterday' => [now()->subDay()->toDateString(), now()->subDay()->toDateString()],
   848→            default => [$today, $today],
   849→        };
   850→    }
   851→}
