   289→    renderMd(text: string): string {

   296→    // gzip 压缩字符串，返回 "gz:<base64>"
   297→    async compressContext(str: string): Promise<string> {
   298→        if (!str) return str
   299→        // 先去除多余空白，减少压缩前体积
   300→        str = str.replace(/\n{3,}/g, '\n\n').trim()
   301→        const stream = new (window as any).CompressionStream('gzip')
   302→        const writer = stream.writable.getWriter()
   303→        writer.write(new TextEncoder().encode(str))
   304→        writer.close()
   305→        const chunks: Uint8Array[] = []
   306→        const reader = stream.readable.getReader()
   307→        while (true) {
   308→            const { done, value } = await reader.read()
   309→            if (done) break
   310→            chunks.push(value)
   311→        }
   312→        const merged = new Uint8Array(chunks.reduce((n, c) => n + c.length, 0))
   313→        let offset = 0
   314→        for (const c of chunks) { merged.set(c, offset); offset += c.length }
   315→        let binary = ''
   316→        for (let i = 0; i < merged.length; i++) binary += String.fromCharCode(merged[i])
   317→        return 'gz:' + btoa(binary)
   318→    }

   509→    private async sendExtractSlots(text: string) {
   510→        this.messages.push({ role: 'user', content: text })
   511→        this.pendingSlotMessages.push({ role: 'user', content: text })
   512→        this.inputText = ''
   513→        this.$nextTick(this.scrollToBottom)
   514→
   515→        const assistantIdx = this.messages.length
   516→        this.messages.push({ role: 'assistant', content: '' })
   517→        this.isStreaming = true
   518→
   519→        try {
   520→            // 收集可用分组名和所有成员花名
   521→            const availableGroups = this.groupOptions.map(g => g.label)
   522→            const availableMembers = [...new Set(this.groupOptions.flatMap(g => g.value))].filter(Boolean)
   523→            // 续查模式：传入上次的槽位结果
   524→            const previousContext = this.lastSlotResult ? {
   525→                member_type: this.lastSlotResult.member_type,
   526→                members: this.lastSlotResult.members,
   527→                period_start: this.lastSlotResult.period_start,
   528→                period_end: this.lastSlotResult.period_end,
   529→            } : undefined
   530→            const hasExistingContext = this.messages.some(m => m.role === 'context')
   531→            const result = await dailyReportAiApis.extractSlots({
   532→                message: text,
   533→                conversation_id: this.slotsConversationId || undefined,
   534→                has_existing_context: hasExistingContext || undefined,
   535→                available_groups: availableGroups,
   536→                available_members: availableMembers,
   537→                previous_context: previousContext,
   538→            })
   539→
   540→            if (!result) {
   541→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   542→                this.pendingSlotMessages.push({ role: 'assistant', content: this.messages[assistantIdx].content })
   543→                await this.flushSlotMessages()
   544→                return
   545→            }
   546→
   547→            // 保存 conversation_id 用于多轮
   548→            this.slotsConversationId = result.conversation_id
   549→
   550→            // 意图为分析已有数据 → 直接走问答，不重新取数据
   551→            if (result.intent === 'analyze') {
   552→                // 移除占位的 assistant 消息，sendAskMessage 会重新创建
   553→                this.messages.splice(assistantIdx, 1)
   554→                this.isStreaming = false
   555→                await this.flushSlotMessages()
   556→                await this.sendAskMessage(text, true)
   557→                return
   558→            }
   559→
   560→            if (!result.slots_complete) {
   561→                // 槽位不完整，显示反问
   562→                this.messages[assistantIdx].content = result.reply
   563→                this.pendingSlotMessages.push({ role: 'assistant', content: result.reply })
   564→                await this.flushSlotMessages()
   565→                return
   566→            }
   567→
   568→            // 槽位完整，保存结果用于续查
   569→            this.lastSlotResult = result
   570→            // 回显意图识别结果
   571→            const nicknames = this.resolveMembers(result)
   572→            const memberLabel = result.member_type === 'group'
   573→                ? `分组：${result.members.join('、')}`
   574→                : `成员：${result.members.join('、')}`
   575→            const countLabel = nicknames.length >= 10 ? `，共 ${nicknames.length} 人` : ''
   576→            const periodLabel = `${result.period_start} ~ ${result.period_end}`
   577→            const slotSummary = `查询对象：${memberLabel}${countLabel}\n时间范围：${periodLabel}${result.period_raw ? `（${result.period_raw}）` : ''}`
   578→            this.messages[assistantIdx].content = slotSummary + '\n\n正在同步日报数据...'
   579→            this.pendingSlotMessages.push({ role: 'assistant', content: slotSummary })
   580→            this.$nextTick(this.scrollToBottom)
   581→
   582→            if (!nicknames.length) {
   583→                const errMsg = `未找到匹配的成员「${result.members.join('、')}」，请尝试使用花名或先手动同步数据。`
   584→                this.messages[assistantIdx].content = errMsg
   585→                this.pendingSlotMessages.push({ role: 'assistant', content: errMsg })
   586→                await this.flushSlotMessages()
   587→                return
   588→            }
   589→
   590→            const context = await this.fetchReportContext(nicknames, result.period_start, result.period_end)
   591→            if (!context) {
   592→                this.messages[assistantIdx].content = '该时间段内没有找到日报数据。'
   593→                this.pendingSlotMessages.push({ role: 'assistant', content: '该时间段内没有找到日报数据。' })
   594→                await this.flushSlotMessages()
   595→                return
   596→            }
   597→
   598→            // 重置意图识别状态
   599→            this.slotsConversationId = ''
   600→
   601→            // 意图识别消息先落库（会创建 session）
   602→            await this.flushSlotMessages()
   603→
   604→            // 保留意图识别结果消息，走正常 ask 流程
   605→            this.reportContext = context
   606→            // 意图已由 extract-slots 处理，ask 只需分析同步好的日报数据
   607→            await this.sendAskMessage('请根据以上日报数据进行分析')
   608→        } catch (e: any) {
   609→            this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   610→            this.pendingSlotMessages.push({ role: 'assistant', content: this.messages[assistantIdx].content })
   611→            await this.flushSlotMessages()
   612→        } finally {
   613→            this.isStreaming = false
   614→        }
   615→    }

   617→    /**
   618→     * 根据槽位结果解析出 nicknames 列表
   619→     */
   620→    private resolveMembers(result: ExtractSlotsResult): string[] {
   621→        if (result.member_type === 'group') {
   622→            // 匹配 groupOptions 的 label，展开为花名列表
   623→            const matched = this.groupOptions.filter(g =>
   624→                result.members.some(m => g.label.includes(m) || m.includes(g.label))
   625→            )
   626→            return [...new Set(matched.flatMap(g => g.value))].filter(Boolean)
   627→        }
   628→        // individual：LLM 返回的成员名必须在花名列表中匹配到，匹配不到则丢弃
   629→        const allMembers = this.groupOptions.flatMap(g => g.value).filter(Boolean)
   630→        return result.members
   631→            .map(name => allMembers.find(m => m.includes(name) || name.includes(m)))
   632→            .filter((n): n is string => !!n)
   633→    }

   635→    /**
   636→     * 根据 nicknames + 日期范围拉取日报数据，返回聚合后的文本上下文
   637→     */
   638→    private async fetchReportContext(nicknames: string[], startDate: string, endDate: string): Promise<string> {
   639→        const days = moment(endDate).diff(moment(startDate), 'day') + 1
   640→        const perPage = Math.max(200, days + 10)
   641→
   642→        const results = await Promise.all(
   643→            nicknames.map(nickname =>
   644→                dailyReportApis.getOtherReportsAPI({
   645→                    start_date: startDate,
   646→                    end_date: endDate,
   647→                    nickname,
   648→                    per_page: perPage,
   649→                    page: 1,
   650→                    show_rest: 1,
   651→                    sys_site: this.sysSite,
   652→                } as any)
   653→            )
   654→        )
   655→
   656→        const dateMap: Record<string, string[]> = {}
   657→        results.forEach((res, idx) => {
   658→            if (!res) return
   659→            const name = nicknames[idx].replace(/\[.+]/, '')
   660→            ;(res.reports ?? []).forEach((item: any) => {
   661→                const date = item.date ?? ''
   662→                if (!dateMap[date]) dateMap[date] = []
   663→                if (item.report) {
   664→                    dateMap[date].push(`${name}：\n${item.report.content ?? ''}`)
   665→                } else if ([4, 5].includes(item.type_id)) {
   666→                    dateMap[date].push(`${name}：休息日`)
   667→                } else {
   668→                    dateMap[date].push(`${name}：未提交`)
   669→                }
   670→            })
   671→        })
   672→
   673→        const sortedDates = Object.keys(dateMap).sort((a, b) => b.localeCompare(a))
   674→        return sortedDates
   675→            .map(date => `【${date}】\n${dateMap[date].join('\n\n')}`)
   676→            .join('\n\n---\n\n')
   677→    }

   679→    /**
   680→     * 正常问答流程：带 reportContext 调 askAi（SSE 流式）
   681→     */
   682→    private async sendAskMessage(text: string, skipUserPush = false) {
   683→        const contextToSend = this.reportContext
   684→        if (contextToSend) {
   685→            const contextLimit = 5000
   686→            const isTruncated = contextToSend.length > contextLimit
   687→            this.messages.push({
   688→                role: 'context',
   689→                content: isTruncated ? contextToSend.slice(0, contextLimit) : contextToSend,
   690→                _truncated: isTruncated,
   691→                _fullLength: contextToSend.length,
   692→                _fullContent: isTruncated ? contextToSend : undefined,
   693→            })
   694→            this.reportContext = ''
   695→        }
   696→        if (!skipUserPush) {
   697→            this.messages.push({ role: 'user', content: text })
   698→            this.inputText = ''
   699→        }
   700→        this.$nextTick(this.scrollToBottom)
   701→
   702→        const assistantIdx = this.messages.length
   703→        this.messages.push({ role: 'assistant', content: '' })
   704→        this.isStreaming = true
   705→
   706→        this.streamAbort = new AbortController()
   707→        try {
   708→            const history = this.messages.slice(0, -2).filter(m => m.role !== 'context' && !!m.content).slice(-10) as { role: 'user' | 'assistant'; content: string }[]
   709→            const processedContext = contextToSend ? compressDailyContext(contextToSend) : ''
   710→            if (processedContext) {
   711→                console.log('[AiChat] 压缩前上下文:\n', processedContext)
   712→            }
   713→            const compressedContext = processedContext ? await this.compressContext(processedContext) : ''
   714→            const response = await dailyReportAiApis.askAi({
   715→                role_id: this.currentRole!.id,
   716→                session_id: this.currentSessionId,
   717→                message: text,
   718→                report_context: compressedContext,
   719→                report_context_raw: contextToSend ? await this.compressContext(contextToSend) : undefined,
   720→                history,
   721→            }, this.streamAbort.signal)
   722→
   723→            const reader = response.body!.getReader()
   724→            const decoder = new TextDecoder()
   725→            let buffer = ''
   726→
   727→            while (true) {
   728→                const { done, value } = await reader.read()
   729→                if (done) break
   730→                buffer += decoder.decode(value, { stream: true })
   731→                const lines = buffer.split('\n')
   732→                buffer = lines.pop() ?? ''
   733→
   734→                for (const line of lines) {
   735→                    if (!line.startsWith('data: ')) continue
   736→                    try {
   737→                        const parsed = JSON.parse(line.slice(6))
   738→                        if (parsed.error_code !== 0) {
   739→                            this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   740→                            break
   741→                        }
   742→                        const data = parsed.data
   743→                        if (data.type === 'session_id') {
   744→                            if (!this.currentSessionId) {
   745→                                this.currentSessionId = Number(data.session_id)
   746→                                await this.loadSessions()
   747→                            }
   748→                        } else if (data.type === 'chat' && data.content) {
   749→                            this.messages[assistantIdx].content += data.content
   750→                            this.$nextTick(this.scrollToBottom)
   751→                        }
   752→                    } catch (_) {}
   753→                }
   754→            }
   755→        } catch (e: any) {
   756→            if (e?.name !== 'AbortError') {
   757→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   758→            }
   759→        } finally {
   760→            if (!this.messages[assistantIdx]?.content && this.streamAbort && !this.streamAbort.signal.aborted) {
   761→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   762→            }
   763→            this.streamAbort = null
   764→            this.isStreaming = false
   765→            // 错误消息落库：后端可能未保存（网络中断、前端超时等），前端兜底
   766→            const finalContent = this.messages[assistantIdx]?.content
   767→            const errorMsg = this.$t('aiChat.errorMsg') as string
   768→            if (finalContent === errorMsg && this.currentSessionId) {
   769→                await dailyReportAiApis.saveMessages({
   770→                    role_id: this.currentRole!.id,
   771→                    session_id: this.currentSessionId,
   772→                    messages: [{ role: 'assistant', content: finalContent }],
   773→                })
   774→            }
   775→            await this.loadSessions()
   776→        }
   777→    }

   779→    // 加载更多 context 内容
   780→    async loadMoreContext(m: { content: string; _id?: number; _truncated?: boolean; _fullLength?: number; _fullContent?: string; _loading?: boolean }) {
   781→        if (!m._truncated || m._loading) return
   782→        const chunkSize = 5000
   783→
   784→        // 实时场景：完整内容在本地 _fullContent 中，直接截取
   785→        if (m._fullContent) {
   786→            const offset = m.content.length
   787→            const nextChunk = m._fullContent.slice(offset, offset + chunkSize)
   788→            m.content += nextChunk
   789→            m._truncated = (offset + chunkSize) < m._fullContent.length
   790→            if (!m._truncated) m._fullContent = undefined // 加载完毕，释放内存
   791→            return
   792→        }
   793→
   794→        // 历史场景：从后端分段加载
   795→        if (!m._id) return
   796→        this.$set(m, '_loading', true)
   797→        try {
   798→            const offset = m.content.length
   799→            const result = await dailyReportAiApis.getContextMore(m._id, offset)
   800→            if (result) {
   801→                m.content += result.content
   802→                m._truncated = result.has_more
   803→                m._fullLength = result.full_length
   804→            }
   805→        } finally {
   806→            this.$set(m, '_loading', false)
   807→        }
   808→    }
