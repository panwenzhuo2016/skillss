+namespace App\Jobs\Notification;
+

+class DailyReportAiEmailJob extends MyJob
+{
+    public function handle(): bool
+    {
+        $now = now();
+        $currentTime = $now->format('H:i');
+        $currentDayOfWeek = $now->dayOfWeekIso; // 1=Monday, 7=Sunday
+        $currentDayOfMonth = $now->day;
+
+        $tasks = DB::table('daily_report_ai_email_tasks')
+            ->where('is_on', 1)
+            ->whereNull('deleted_at')
+            ->where('send_time', $currentTime)
+            ->get();
+
+        if ($tasks->isEmpty()) {
+            return true;
+        }
+
+        foreach ($tasks as $task) {
+            if (!$this->shouldSendNow($task->send_cycle, $currentDayOfWeek, $currentDayOfMonth)) {
+                continue;
+            }
+
+            try {
+                $this->processTask($task);
+            } catch (\Throwable $e) {
+                info("[DailyReportAiEmailJob] task={$task->id} failed: {$e->getMessage()}".$e->getTraceAsString());
+            }
+        }
+
+        info('[DailyReportAiEmailJob] finished');
+        return true;
+    }
+
+    private function shouldSendNow(string $sendCycle, int $dayOfWeek, int $dayOfMonth): bool
+    {
+        return match ($sendCycle) {
+            'daily' => true,
+            'weekly_monday' => $dayOfWeek === 1,
+            'weekly_friday' => $dayOfWeek === 5,
+            'monthly_first' => $dayOfMonth === 1,
+            default => false,
+        };
+    }
+
+    private function processTask(object $task): void
+    {
+        $sfuserId = $task->sfuser_id;
+
+        // 读取邮箱配置
+        $mc = MailConfig::where('type', "daily_report_{$sfuserId}")->first();
+        if (!$mc || empty(Arr::get($mc->config, 'to_email'))) {
+            info("[DailyReportAiEmailJob] task={$task->id} no email config, skip");
+            return;
+        }
+
+        $toEmails = Arr::get($mc->config, 'to_email');
+        $sfuserToken = Arr::get($mc->config, 'sfuser_authorized_token', '');
+        if (!$sfuserToken) {
+            info("[DailyReportAiEmailJob] task={$task->id} no sfuser_token, skip");
+            return;
+        }
+
+        // 确定日报日期范围
+        [$startDate, $endDate] = $this->getDateRange($task->report_range);
+
+        // 拉取日报数据
+        $dailyService = app(DailyReportService::class);
+        $focusUserIds = $dailyService->getSubordinates($sfuserId);
+        $data = $dailyService->getOthersDailyReportListEnable(
+            $startDate,
+            $endDate,
+            $sfuserId,
+            $focusUserIds['focus_users'],
+            10000
+        );
+
+        if (empty($data) || empty($data['reports']) || empty($data['reports']['report_list'])) {
+            info("[DailyReportAiEmailJob] task={$task->id} no report data, skip");
+            return;
+        }
+
+        // 构建邮件内容（加密格式，由 oa-core 解密）
+        $content = '';
+        $no = 0;
+        foreach ($data['reports']['report_list'] as $report) {
+            $nickname = $report['data']['nickname'] ?? '';
+            if (empty($nickname)) {
+                continue;
+            }
+            $nickname = EncryptService::app()->markEncrString($nickname);
+            $reportContent = EncryptService::app()->markEncrString($report['content']);
+            ++$no;
+            $content .= '<p><b>'.$no.'、 '.$nickname.'</b></p>';
+            $content .= '<div style="border:1px solid #eee;padding:8px;margin-bottom:12px;">'.$reportContent.'</div>';
+        }
+
+        $subject = "【{$startDate}】 {$task->name} - AI 分析";
+
+        // 构建邮件参数（手动构建以支持 ai_analysis_config）
+        $params = [
+            'decrypt_config' => [
+                'obj' => '$this',
+                'decrypt_columns' => [],
+                'partly_decrypt_columns' => ['content', 'subject'],
+            ],
+            'receivers' => [[
+                'type' => 'token',
+                'tokens' => [$sfuserToken],
+                'filter' => 'emails',
+                'filter_map' => ['emails' => $toEmails],
+            ]],
+            'content' => $content,
+            'subject' => $subject,
+            'ai_analysis_config' => [
+                'system_prompt' => $task->system_prompt,
+            ],
+        ];
+
+        info("[DailyReportAiEmailJob] task={$task->id} sending email to ".implode(',', $toEmails));
+        CoreApiLib::sendRequest('master-api/mail/send-decr', $params);
+        info("[DailyReportAiEmailJob] task={$task->id} email sent");
+    }
+
+    private function getDateRange(string $reportRange): array
+    {
+        $today = now()->toDateString();
+        return match ($reportRange) {
+            'today' => [$today, $today],
+            'yesterday' => [now()->subDay()->toDateString(), now()->subDay()->toDateString()],
+            default => [$today, $today],
+        };
+    }
