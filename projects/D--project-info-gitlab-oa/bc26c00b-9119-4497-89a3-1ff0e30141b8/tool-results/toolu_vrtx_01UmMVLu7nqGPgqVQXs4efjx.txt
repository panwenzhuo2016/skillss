   436→   436→                setTimeout(() => this.scrollToBottom(), 100)
   437→   437→            })
   438→   438→        } finally {
   439→   439→            this.messagesLoading = false
   440→   440→        }
   441→   441→    }

   450→   450→        this.accumulatedContextChars = 0

   461→   461→    async sendMessage() {
   462→   462→        const text = this.inputText.trim()
   463→   463→        if (!text || !this.currentRole || this.isStreaming) {return}
   464→   464→
   465→   465→        // 有手动同步的 reportContext 且从未做过意图识别 → 直接问答（跳过意图识别）
   466→   466→        if (this.reportContext && !this.lastSlotResult) {
   467→   467→            await this.sendAskMessage(text)
   468→   468→        } else {
   469→   469→            // 走意图识别：LLM 判断是分析已有数据还是查询新数据

   483→   483→    /** 检查累积上下文是否可能超限，返回 'ok' 继续发送 / 'newSession' 新建会话 / 'abort' 关闭弹窗不操作 */
   484→   484→    private async checkContextOverLimit(extraChars = 0): Promise<'ok' | 'newSession' | 'abort'> {
   485→   485→        const contextCharLimit = 2_200_000
   486→   486→        if (this.accumulatedContextChars + extraChars <= contextCharLimit) {return 'ok'}
   487→   487→        try {
   488→   488→            await this.$confirm(
   489→   489→                this.$t('aiChat.contextOverLimit') as string,
   490→   490→                this.$t('aiChat.confirmTitle') as string,
   491→   491→                {
   492→   492→                    confirmButtonText: this.$t('aiChat.contextOverLimitContinue') as string,
   493→   493→                    cancelButtonText: this.$t('aiChat.contextOverLimitNewSession') as string,
   494→   494→                    type: 'warning',
   495→   495→                    distinguishCancelAndClose: true,
   496→   496→                },
   497→   497→            )
   498→   498→            return 'ok' // 点击「继续发送」
   499→   499→        } catch (action) {
   500→   500→            return action === 'cancel' ? 'newSession' : 'abort'
   501→   501→        }
   502→   502→    }

   504→   504→    /** 将 pendingSlotMessages 中最后一条 assistant 的 content 同步为 UI 上的最终值 */
   505→   505→    private syncPendingAssistant(assistantIdx: number) {
   506→   506→        const finalContent = this.messages[assistantIdx]?.content
   507→   507→        if (!finalContent) { return }
   508→   508→        // 从后往前找最后一条 assistant，更新其 content
   509→   509→        for (let i = this.pendingSlotMessages.length - 1; i >= 0; i--) {
   510→   510→            if (this.pendingSlotMessages[i].role === 'assistant') {
   511→   511→                this.pendingSlotMessages[i].content = finalContent
   512→   512→                return
   513→   513→            }
   514→   514→        }
   515→   515→    }

   535→   535→            if (res) {
   536→   536→                this.$store.commit('setDailyReportAllSubordinates', res)
   537→   537→            }
   538→   538→        } catch (e) {
   539→   539→            // 刷新失败不阻塞主流程，使用缓存数据

   541→   541→        }

   544→   544→    private async sendExtractSlots(text: string) {
   545→   545→        const limitAction = await this.checkContextOverLimit()
   546→   546→        if (limitAction === 'newSession') { this.newSession(); return }
   547→   547→        if (limitAction === 'abort') { return }
   548→   548→
   549→   549→        this.messages.push({ role: 'user', content: text })
   550→   550→        this.pendingSlotMessages.push({ role: 'user', content: text })
   551→   551→        this.inputText = ''
   552→   552→        this.$nextTick(this.scrollToBottom)
   553→   553→
   554→   554→        const assistantIdx = this.messages.length
   555→   555→        this.messages.push({ role: 'assistant', content: '' })
   556→   556→        this.isStreaming = true
   557→   557→
   558→   558→        try {
   559→   559→            // 每次意图识别前刷新下属列表，确保分组和成员是最新的
   560→   560→            await this.refreshSubordinates()
   561→   561→            // 收集可用分组名和所有成员花名
   562→   562→            const availableGroups = this.groupOptions.map(g => g.label)
   563→   563→            const availableMembers = [...new Set(this.groupOptions.flatMap(g => g.value))].filter(Boolean)
   564→   564→            // 续查模式：传入上次的槽位结果（含 query_type 和 project_names）
   565→   565→            const previousContext = this.lastSlotResult ? {
   566→   566→                query_type: this.lastSlotResult.query_type,
   567→   567→                member_type: this.lastSlotResult.member_type,
   568→   568→                members: this.lastSlotResult.members,
   569→   569→                project_names: this.lastSlotResult.project_names,
   570→   570→                period_start: this.lastSlotResult.period_start,
   571→   571→                period_end: this.lastSlotResult.period_end,
   572→   572→            } : undefined
   573→   573→            const hasExistingContext = this.messages.some(m => m.role === 'context')
   574→   574→            // 构建用户历史提问（仅用户消息，助手回复太长且无意义，上下文由 previous_context 承载）
   575→   575→            const userHistory = this.messages
   576→   576→                .slice(0, -1) // 排除占位的空 assistant
   577→   577→                .filter(m => m.role === 'user' && m.content)
   578→   578→                .slice(0, -1) // 排除刚加的当前 user 消息
   579→   579→            let messageForSlots = text
   580→   580→            if (userHistory.length > 0) {
   581→   581→                const historyText = userHistory.map(m => `- ${m.content}`).join('\n')
   582→   582→                messageForSlots = `[用户历史提问]\n${historyText}\n\n[当前输入]\n${text}`
   583→   583→            }
   584→   584→            const result = await dailyReportAiApis.extractSlots({
   585→   585→                message: messageForSlots,
   586→   586→                has_existing_context: hasExistingContext || undefined,
   587→   587→                available_groups: availableGroups,
   588→   588→                available_members: availableMembers,
   589→   589→                previous_context: previousContext,
   590→   590→            })
   591→   591→            if (!result) {
   592→   592→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   593→   593→                this.pendingSlotMessages.push({ role: 'assistant', content: this.messages[assistantIdx].content })
   594→   594→                await this.flushSlotMessages()
   595→   595→                return
   596→   596→            }
   597→   597→
   598→   598→            // 保存 conversation_id 用于多轮
   599→   599→            this.slotsConversationId = result.conversation_id
   600→   600→
   601→   601→            // 意图为分析已有数据 → 从历史 context 消息恢复数据，直接走问答
   602→   602→            if (result.intent === 'analyze') {
   603→   603→                // reportContext 每次 sendAskMessage 后会被清空，需要从 messages 中恢复
   604→   604→                const lastCtxMsg = [...this.messages].reverse().find(m => m.role === 'context')
   605→   605→                if (lastCtxMsg) {
   606→   606→                    this.reportContext = lastCtxMsg._fullContent || lastCtxMsg.content
   607→   607→                }
   608→   608→                this.messages.splice(assistantIdx, 1)
   609→   609→                this.isStreaming = false
   610→   610→                await this.flushSlotMessages()
   611→   611→                await this.sendAskMessage(text, true)
   612→   612→                return
   613→   613→            }
   614→   614→
   615→   615→            if (!result.slots_complete) {
   616→   616→                this.messages[assistantIdx].content = result.reply
   617→   617→                this.pendingSlotMessages.push({ role: 'assistant', content: result.reply })
   618→   618→                await this.flushSlotMessages()
   619→   619→                return
   620→   620→            }
   621→   621→
   622→   622→            // 槽位完整，保存结果用于续查
   623→   623→            this.lastSlotResult = result
   624→   624→
   625→   625→            // 根据 query_type 分发到不同处理流程
   626→   626→            switch (result.query_type) {
   627→   627→                case 'project':
   628→   628→                    await this.handleProjectQuery(result, assistantIdx, text)
   629→   629→                    break
   630→   630→                case 'person':
   631→   631→                    await this.handlePersonQuery(result, assistantIdx, text)
   632→   632→                    break
   633→   633→                case 'global':
   634→   634→                    await this.handleGlobalQuery(result, assistantIdx, text)
   635→   635→                    break
   636→   636→                default:
   637→   637→                    // unknown：如果有 members 走 person，否则走 global
   638→   638→                    if (result.members.length > 0) {
   639→   639→                        await this.handlePersonQuery(result, assistantIdx, text)
   640→   640→                    } else {
   641→   641→                        await this.handleGlobalQuery(result, assistantIdx, text)
   642→   642→                    }
   643→   643→                    break
   644→   644→            }
   645→   645→        } catch (e) {
   646→   646→            console.error('[AiChat] sendExtractSlots error:', e)
   647→   647→            this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   648→   648→            this.pendingSlotMessages.push({ role: 'assistant', content: this.messages[assistantIdx].content })
   649→   649→            await this.flushSlotMessages()
   650→   650→        } finally {
   651→   651→            this.isStreaming = false
   652→   652→        }
   653→   653→    }

   659→   659→    private async handleProjectQuery(result: ExtractSlotsResult, assistantIdx: number, userText: string) {
   660→   660→        const projectLabel = `项目：${result.project_names.join('、')}`
   661→   661→        const periodLabel = `${result.period_start} ~ ${result.period_end}`
   662→   662→        const slotSummary = `查询类型：项目进度\n${projectLabel}\n时间范围：${periodLabel}${result.period_raw ? `（${result.period_raw}）` : ''}`
   663→   663→        this.messages[assistantIdx].content = slotSummary + '\n\n正在查询项目关联人员...'
   664→   664→        this.pendingSlotMessages.push({ role: 'assistant', content: slotSummary })
   665→   665→        this.$nextTick(this.scrollToBottom)
   666→   666→

   668→   668→        const projectResults = await dailyReportAiApis.resolveProjectMembers(result.project_names)
   669→   669→        const allNicknames = [...new Set(projectResults.flatMap(p => p.nicknames))].filter(Boolean)
   670→   670→        const hasRoomHit = projectResults.some(p => p.room_id && p.nicknames.length > 0)
   671→   671→
   672→   672→        if (hasRoomHit && allNicknames.length > 0) {
   673→   673→            // 索引命中：用关联人员的花名拉原文日报
   674→   674→            this.messages[assistantIdx].content = slotSummary + `\n\n已找到 ${allNicknames.length} 名关联人员，正在同步日报数据...`
   675→   675→            this.$nextTick(this.scrollToBottom)
   676→   676→
   677→   677→            const context = await this.fetchReportContext(allNicknames, result.period_start, result.period_end)
   678→   678→            if (!context) {
   679→   679→                this.messages[assistantIdx].content = slotSummary + '\n\n该项目在指定时间段内无日报更新。'
   680→   680→                this.syncPendingAssistant(assistantIdx)
   681→   681→                await this.flushSlotMessages()
   682→   682→                return
   683→   683→            }
   684→   684→
   685→   685→            this.slotsConversationId = ''
   686→   686→            this.syncPendingAssistant(assistantIdx)
   687→   687→            await this.flushSlotMessages()
   688→   688→            this.reportContext = context
   689→   689→            await this.sendAskMessage(
   690→   690→                `请分析以上日报数据中关于项目「${result.project_names.join('、')}」的进展情况。`,
   691→   691→                false, 'project',
   692→   692→            )
   693→   693→        } else {
   694→   694→            // 索引未命中：降级走全局摘要 + 项目关键词
   695→   695→            this.messages[assistantIdx].content = slotSummary + '\n\n项目索引未命中，正在通过周摘要检索...'
   696→   696→            this.$nextTick(this.scrollToBottom)
   697→   697→            await this.handleGlobalQueryWithKeywords(result, assistantIdx, result.project_names, userText)
   698→   698→        }
   699→   699→    }

   763→   763→    /**
   764→   764→     * 全局查询核心逻辑：获取周摘要 → 拼接 → 喂给 AI
   765→   765→     * @param keywords 项目降级时的关键词，空数组表示无特定项目
   766→   766→     */
   767→   767→    private async handleGlobalQueryWithKeywords(result: ExtractSlotsResult, assistantIdx: number, keywords: string[], userText: string) {
   768→   768→        // 计算周范围（每个 ISO 周的周一）
   769→   769→        const weekStarts = this.computeWeekStarts(result.period_start, result.period_end)
   770→   770→
   771→   771→        this.messages[assistantIdx].content += `\n查询 ${weekStarts.length} 周的周摘要...`
   772→   772→        this.$nextTick(this.scrollToBottom)
   773→   773→
   774→   774→        // 批量获取周摘要（后端自动取当前用户下属，已解密）
   775→   775→        const summaries = await dailyReportAiApis.batchGetSummaries(weekStarts)
   776→   776→
   777→   777→        if (summaries.length === 0) {
   778→   778→            this.messages[assistantIdx].content += '\n\n该时间段内没有周摘要数据。请确认相关人员是否已生成周摘要。'
   779→   779→            this.syncPendingAssistant(assistantIdx)
   780→   780→            await this.flushSlotMessages()
   781→   781→            return
   782→   782→        }
   783→   783→
   784→   784→        // 拼接摘要为上下文
   785→   785→        const context = summaries
   786→   786→            .map(s => `[${s.week_start}~${s.week_end}] sfuser_id:${s.sfuser_id}\n${s.content}`)
   787→   787→            .join('\n\n---\n\n')
   788→   788→
   789→   789→        this.slotsConversationId = ''
   790→   790→        this.syncPendingAssistant(assistantIdx)
   791→   791→        await this.flushSlotMessages()
   792→   792→        this.reportContext = context
   793→   793→
   794→   794→        // 构建 ask 消息，带上用户原始提问
   795→   795→        const sceneType = keywords.length > 0 ? 'project_fallback' : 'global'
   796→   796→        let askMsg = '请基于以上周摘要数据进行全局聚合分析。'
   797→   797→        if (keywords.length > 0) {
   798→   798→            askMsg = `请基于以上周摘要数据，重点检索与「${keywords.join('、')}」相关的内容进行分析。`
   799→   799→        }
   800→   800→        await this.sendAskMessage(askMsg, false, sceneType)
   801→   801→    }

   803→   803→    /**
   804→   804→     * 计算时间范围内的所有 ISO 周的周一日期
   805→   805→     */
   806→   806→    private computeWeekStarts(startDate: string, endDate: string): string[] {
   807→   807→        const starts: string[] = []
   808→   808→        let current = moment(startDate).startOf('isoWeek')
   809→   809→        const end = moment(endDate)
   810→   810→        while (current.isSameOrBefore(end)) {
   811→   811→            starts.push(current.format('YYYY-MM-DD'))
   812→   812→            current = current.add(1, 'week')
   813→   813→        }
   814→   814→        return starts
   815→   815→    }

   817→   817→    /**
   818→   818→     * 根据槽位结果解析出 nicknames 列表
   819→   819→     */
   820→   820→    private resolveMembers(result: ExtractSlotsResult): string[] {
   821→   821→        if (result.member_type === 'group') {

   918→   918→    private async sendAskMessage(text: string, skipUserPush = false, sceneType?: string) {
   919→   919→        const contextToSend = this.reportContext
   920→   920→        // 记录本轮消息插入起点，取消时用于回退
   921→   921→        const msgStartIdx = this.messages.length
   922→   922→        if (contextToSend) {
   923→   923→            const contextLimit = 5000
   924→   924→            const isTruncated = contextToSend.length > contextLimit
   925→   925→            this.messages.push({
   926→   926→                role: 'context',
   927→   927→                content: isTruncated ? contextToSend.slice(0, contextLimit) : contextToSend,
   928→   928→                _truncated: isTruncated,
   929→   929→                _fullLength: contextToSend.length,
   930→   930→                _fullContent: isTruncated ? contextToSend : undefined,
   931→   931→            })
   932→   932→            this.reportContext = ''
   933→   933→        }
   934→   934→        if (!skipUserPush) {
   935→   935→            this.messages.push({ role: 'user', content: text })
   936→   936→            this.inputText = ''
   937→   937→        }
   938→   938→        this.$nextTick(this.scrollToBottom)
   939→   939→
   940→   940→        const assistantIdx = this.messages.length
   941→   941→        this.messages.push({ role: 'assistant', content: '' })
   942→   942→        this.isStreaming = true
   943→   943→
   944→   944→        const newContextChars = contextToSend ? contextToSend.length : 0
   945→   945→        const limitAction = await this.checkContextOverLimit(newContextChars)
   946→   946→        if (limitAction !== 'ok') {
   947→   947→            this.messages.splice(msgStartIdx)
   948→   948→            this.isStreaming = false
   949→   949→            if (limitAction === 'newSession') { this.newSession() }
   950→   950→            return
   951→   951→        }
   952→   952→
   953→   953→        this.streamAbort = new AbortController()
   954→   954→        try {
   955→   955→            const history = this.messages.slice(0, -2).filter(m => m.role !== 'context' && !!m.content).slice(-10) as { role: 'user' | 'assistant'; content: string }[]
   956→   956→            const processedContext = contextToSend ? compressDailyContext(contextToSend) : ''
   957→   957→            if (processedContext) {
   958→   958→                console.info('[AiChat] 压缩前上下文:\n', processedContext)
   959→   959→            }
   960→   960→            const compressedContext = processedContext ? await compressContext(processedContext) : ''
   961→   961→            const response = await dailyReportAiApis.askAi({
   962→   962→                role_id: this.currentRole!.id,
   963→   963→                session_id: this.currentSessionId,
   964→   964→                message: text,
   965→   965→                report_context: compressedContext,
   966→   966→                report_context_raw: contextToSend ? await compressContext(contextToSend) : undefined,
   967→   967→                system_prompt: this.currentRole!.system_prompt
   968→   968→                    + (sceneType && this.SCENE_OUTPUT_RULES[sceneType] || ''),
   969→   969→                history,
   970→   970→            }, this.streamAbort.signal)
   971→   971→
   972→   972→            const reader = response.body!.getReader()
   973→   973→            const decoder = new TextDecoder()
   974→   974→            let buffer = ''
   975→   975→
   976→   976→            while (true) {
   977→   977→                const { done, value } = await reader.read()
   978→   978→                if (done) {break}
   979→   979→                buffer += decoder.decode(value, { stream: true })
   980→   980→                const lines = buffer.split('\n')
   981→   981→                buffer = lines.pop() ?? ''
   982→   982→
   983→   983→                for (const line of lines) {
   984→   984→                    if (!line.startsWith('data: ')) {continue}
   985→   985→                    try {
   986→   986→                        const parsed = JSON.parse(line.slice(6))
   987→   987→                        if (parsed.error_code !== 0) {
   988→   988→                            this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   989→   989→                            break
   990→   990→                        }
   991→   991→                        const data = parsed.data
   992→   992→                        // TransferRequestService SSE 格式：meta（首帧，含 session_id）+ chat（流式内容）
   993→   993→                        if (data.type === 'meta') {
   994→   994→                            if (data.session_id && !this.currentSessionId) {
   995→   995→                                this.currentSessionId = Number(data.session_id)
   996→   996→                                this.loadSessions()
   997→   997→                            }
   998→   998→                        } else if (data.type === 'chat' && data.content) {
   999→   999→                            this.messages[assistantIdx].content += data.content
  1000→  1000→                            this.$nextTick(this.scrollToBottom)
  1001→  1001→                        }
  1002→  1002→                    } catch {}
  1003→  1003→                }
  1004→  1004→            }
  1005→  1005→            // 流式读取完成，累加本次上下文字符数
  1006→  1006→            if (newContextChars > 0) {
  1007→  1007→                this.accumulatedContextChars += newContextChars
  1008→  1008→            }
  1009→  1009→        } catch (e: any) {
  1010→  1010→            if (e?.name !== 'AbortError') {

  1012→  1012→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
  1013→  1013→            }
  1014→  1014→        } finally {
  1015→  1015→            if (!this.messages[assistantIdx]?.content && this.streamAbort && !this.streamAbort.signal.aborted) {
  1016→  1016→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
  1017→  1017→            }
  1018→  1018→            this.streamAbort = null
  1019→  1019→            this.isStreaming = false
  1020→  1020→            // 落库：仅保存 AI 回复（context 和 user 消息已由后端 ask 接口保存，避免重复）
  1021→  1021→            const assistantContent = this.messages[assistantIdx]?.content
  1022→  1022→            if (assistantContent && this.currentSessionId) {
  1023→  1023→                await dailyReportAiApis.saveMessages({
  1024→  1024→                    role_id: this.currentRole!.id,
  1025→  1025→                    session_id: this.currentSessionId,
  1026→  1026→                    messages: [{ role: 'assistant', content: assistantContent }],
  1027→  1027→                })
  1028→  1028→            }
  1029→  1029→            await this.loadSessions()
  1030→  1030→        }
  1031→  1031→    }

  1034→  1034→    async loadMoreContext(m: { content: string; _id?: number; _truncated?: boolean; _fullLength?: number; _fullContent?: string; _loading?: boolean }) {
  1035→  1035→        if (!m._truncated || m._loading) {return}

  1038→  1038→        // 实时场景：完整内容在本地 _fullContent 中，直接截取

  1041→  1041→            const nextChunk = m._fullContent.slice(offset, offset + chunkSize)
  1042→  1042→            m.content += nextChunk
  1043→  1043→            m._truncated = (offset + chunkSize) < m._fullContent.length
  1044→  1044→            if (!m._truncated) {m._fullContent = undefined} // 加载完毕，释放内存
  1045→  1045→            return
  1046→  1046→        }
