   504→    /** 将 pendingSlotMessages 中最后一条 assistant 的 content 同步为 UI 上的最终值 */

   512→                return
   513→            }
   514→        }

   517→    private async flushSlotMessages() {
   518→        if (this.pendingSlotMessages.length === 0 || !this.currentRole) {return}

   525→            this.currentSessionId = sessionId
   526→            await this.loadSessions()
   527→        }
   528→        this.pendingSlotMessages = []
   529→    }

   544→    private async sendExtractSlots(text: string) {
   545→        const limitAction = await this.checkContextOverLimit()
   546→        if (limitAction === 'newSession') { this.newSession(); return }
   547→        if (limitAction === 'abort') { return }
   548→
   549→        this.messages.push({ role: 'user', content: text })
   550→        this.pendingSlotMessages.push({ role: 'user', content: text })
   551→        this.inputText = ''
   552→        this.$nextTick(this.scrollToBottom)
   553→
   554→        const assistantIdx = this.messages.length
   555→        this.messages.push({ role: 'assistant', content: '' })
   556→        this.isStreaming = true
   557→
   558→        try {
   559→            // 每次意图识别前刷新下属列表，确保分组和成员是最新的
   560→            await this.refreshSubordinates()
   561→            // 收集可用分组名和所有成员花名
   562→            const availableGroups = this.groupOptions.map(g => g.label)
   563→            const availableMembers = [...new Set(this.groupOptions.flatMap(g => g.value))].filter(Boolean)
   564→            // 续查模式：传入上次的槽位结果（含 query_type 和 project_names）
   565→            const previousContext = this.lastSlotResult ? {
   566→                query_type: this.lastSlotResult.query_type,
   567→                member_type: this.lastSlotResult.member_type,
   568→                members: this.lastSlotResult.members,
   569→                project_names: this.lastSlotResult.project_names,
   570→                period_start: this.lastSlotResult.period_start,
   571→                period_end: this.lastSlotResult.period_end,
   572→            } : undefined
   573→            const hasExistingContext = this.messages.some(m => m.role === 'context')
   574→            // 构建用户历史提问（仅用户消息，助手回复太长且无意义，上下文由 previous_context 承载）
   575→            const userHistory = this.messages
   576→                .slice(0, -1) // 排除占位的空 assistant
   577→                .filter(m => m.role === 'user' && m.content)
   578→                .slice(0, -1) // 排除刚加的当前 user 消息
   579→            let messageForSlots = text
   580→            if (userHistory.length > 0) {
   581→                const historyText = userHistory.map(m => `- ${m.content}`).join('\n')
   582→                messageForSlots = `[用户历史提问]\n${historyText}\n\n[当前输入]\n${text}`
   583→            }
   584→            const result = await dailyReportAiApis.extractSlots({
   585→                message: messageForSlots,
   586→                has_existing_context: hasExistingContext || undefined,
   587→                available_groups: availableGroups,
   588→                available_members: availableMembers,
   589→                previous_context: previousContext,
   590→            })
   591→            if (!result) {
   592→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   593→                this.pendingSlotMessages.push({ role: 'assistant', content: this.messages[assistantIdx].content })
   594→                await this.flushSlotMessages()
   595→                return
   596→            }
   597→
   598→            // 保存 conversation_id 用于多轮
   599→            this.slotsConversationId = result.conversation_id
   600→
   601→            // 意图为分析已有数据 → 从历史 context 消息恢复数据，直接走问答
   602→            if (result.intent === 'analyze') {
   603→                // reportContext 每次 sendAskMessage 后会被清空，需要从 messages 中恢复
   604→                const lastCtxMsg = [...this.messages].reverse().find(m => m.role === 'context')
   605→                if (lastCtxMsg) {
   606→                    this.reportContext = lastCtxMsg._fullContent || lastCtxMsg.content
   607→                }
   608→                this.messages.splice(assistantIdx, 1)
   609→                this.isStreaming = false
   610→                await this.flushSlotMessages()
   611→                await this.sendAskMessage(text, true)
   612→                return
   613→            }
   614→
   615→            if (!result.slots_complete) {
   616→                this.messages[assistantIdx].content = result.reply
   617→                this.pendingSlotMessages.push({ role: 'assistant', content: result.reply })
   618→                await this.flushSlotMessages()
   619→                return
   620→            }
   621→
   622→            // 槽位完整，保存结果用于续查
   623→            this.lastSlotResult = result
   624→
   625→            // 根据 query_type 分发到不同处理流程
   626→            switch (result.query_type) {
   627→                case 'project':
   628→                    await this.handleProjectQuery(result, assistantIdx, text)
   629→                    break
   630→                case 'person':
   631→                    await this.handlePersonQuery(result, assistantIdx, text)
   632→                    break
   633→                case 'global':
   634→                    await this.handleGlobalQuery(result, assistantIdx, text)
   635→                    break
   636→                default:
   637→                    // unknown：如果有 members 走 person，否则走 global
   638→                    if (result.members.length > 0) {
   639→                        await this.handlePersonQuery(result, assistantIdx, text)
   640→                    } else {
   641→                        await this.handleGlobalQuery(result, assistantIdx, text)
   642→                    }
   643→                    break
   644→            }
   645→        } catch (e) {
   646→            console.error('[AiChat] sendExtractSlots error:', e)
   647→            this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   648→            this.pendingSlotMessages.push({ role: 'assistant', content: this.messages[assistantIdx].content })
   649→            await this.flushSlotMessages()
   650→        } finally {
   651→            this.isStreaming = false
   652→        }
   653→    }

   659→    private async handleProjectQuery(result: ExtractSlotsResult, assistantIdx: number, userText: string) {
   660→        const projectLabel = `项目：${result.project_names.join('、')}`
   661→        const periodLabel = `${result.period_start} ~ ${result.period_end}`
   662→        const slotSummary = `查询类型：项目进度\n${projectLabel}\n时间范围：${periodLabel}${result.period_raw ? `（${result.period_raw}）` : ''}`
   663→        this.messages[assistantIdx].content = slotSummary + '\n\n正在查询项目关联人员...'
   664→        this.pendingSlotMessages.push({ role: 'assistant', content: slotSummary })

   668→        const projectResults = await dailyReportAiApis.resolveProjectMembers(result.project_names)
   669→        const allNicknames = [...new Set(projectResults.flatMap(p => p.nicknames))].filter(Boolean)
   670→        const hasRoomHit = projectResults.some(p => p.room_id && p.nicknames.length > 0)
   671→
   672→        if (hasRoomHit && allNicknames.length > 0) {
   673→            // 索引命中：用关联人员的花名拉原文日报
   674→            this.messages[assistantIdx].content = slotSummary + `\n\n已找到 ${allNicknames.length} 名关联人员，正在同步日报数据...`

   678→            if (!context) {
   679→                this.messages[assistantIdx].content = slotSummary + '\n\n该项目在指定时间段内无日报更新。'
   680→                this.syncPendingAssistant(assistantIdx)
   681→                await this.flushSlotMessages()
   682→                return
   683→            }
   684→
   685→            this.slotsConversationId = ''
   686→            this.syncPendingAssistant(assistantIdx)
   687→            await this.flushSlotMessages()
   688→            this.reportContext = context
   689→            await this.sendAskMessage(
   690→                `请分析以上日报数据中关于项目「${result.project_names.join('、')}」的进展情况。`,
   691→                false, 'project',
   692→            )
   693→        } else {
   694→            // 索引未命中：降级走全局摘要 + 项目关键词
   695→            this.messages[assistantIdx].content = slotSummary + '\n\n项目索引未命中，正在通过周摘要检索...'
   696→            this.$nextTick(this.scrollToBottom)
   697→            await this.handleGlobalQueryWithKeywords(result, assistantIdx, result.project_names, userText)
   698→        }
   699→    }

   799→        }

   801→    }

   803→    /**
   804→     * 计算时间范围内的所有 ISO 周的周一日期
   805→     */
   806→    private computeWeekStarts(startDate: string, endDate: string): string[] {
   807→        const starts: string[] = []
   808→        let current = moment(startDate).startOf('isoWeek')
   809→        const end = moment(endDate)
   810→        while (current.isSameOrBefore(end)) {
   811→            starts.push(current.format('YYYY-MM-DD'))
   812→            current = current.add(1, 'week')
   813→        }
   814→        return starts
   815→    }

   817→    /**
   818→     * 根据槽位结果解析出 nicknames 列表
   819→     */
   820→    private resolveMembers(result: ExtractSlotsResult): string[] {
   821→        if (result.member_type === 'group') {
   822→            // 匹配 groupOptions 的 label，展开为花名列表
   823→            const matched = this.groupOptions.filter(g =>
   824→                result.members.some(m => g.label.includes(m) || m.includes(g.label))
   825→            )
   826→            return [...new Set(matched.flatMap(g => g.value))].filter(Boolean)
   827→        }
   828→        // individual：LLM 返回的成员名必须在花名列表中匹配到，匹配不到则丢弃
   829→        const allMembers = this.groupOptions.flatMap(g => g.value).filter(Boolean)
   830→        return result.members
   831→            .map(name => allMembers.find(m => m.includes(name) || name.includes(m)))
   832→            .filter((n): n is string => !!n)
   833→    }

   835→    /**
   836→     * 根据 nicknames + 日期范围拉取日报数据，返回聚合后的文本上下文
   837→     */
   838→    private async fetchReportContext(nicknames: string[], startDate: string, endDate: string): Promise<string> {
   839→        const days = moment(endDate).diff(moment(startDate), 'day') + 1
   840→        const perPage = Math.max(200, days + 10)
   841→
   842→        const results = await Promise.all(
   843→            nicknames.map(nickname =>
   844→                dailyReportApis.getOtherReportsAPI({
   845→                    start_date: startDate,
   846→                    end_date: endDate,
   847→                    nickname,
   848→                    per_page: perPage,
   849→                    page: 1,
   850→                    show_rest: 1,
   851→                    sys_site: this.sysSite,
   852→                } as any)
   853→            )
   854→        )
   855→
   856→        const dateMap: Record<string, string[]> = {}
   857→        results.forEach((res, idx) => {
   858→            if (!res) {return}
   859→            const name = nicknames[idx].replace(/\[.+]/, '')
   860→            ;(res.reports ?? []).forEach((item: any) => {
   861→                const date = item.date ?? ''
   862→                if (!dateMap[date]) {dateMap[date] = []}
   863→                if (item.report) {
   864→                    dateMap[date].push(`${name}：\n${item.report.content ?? ''}`)
   865→                } else if ([4, 5].includes(item.type_id)) {
   866→                    dateMap[date].push(`${name}：休息日`)
   867→                } else {
   868→                    dateMap[date].push(`${name}：未提交`)
   869→                }
   870→            })
   871→        })
   872→
   873→        const sortedDates = Object.keys(dateMap).sort((a, b) => b.localeCompare(a))
   874→        return sortedDates
   875→            .map(date => `【${date}】\n${dateMap[date].join('\n\n')}`)
   876→            .join('\n\n---\n\n')
   877→    }

   918→    private async sendAskMessage(text: string, skipUserPush = false, sceneType?: string) {
   919→        const contextToSend = this.reportContext
   920→        // 记录本轮消息插入起点，取消时用于回退
   921→        const msgStartIdx = this.messages.length
   922→        if (contextToSend) {
   923→            const contextLimit = 5000
   924→            const isTruncated = contextToSend.length > contextLimit
   925→            this.messages.push({
   926→                role: 'context',
   927→                content: isTruncated ? contextToSend.slice(0, contextLimit) : contextToSend,
   928→                _truncated: isTruncated,
   929→                _fullLength: contextToSend.length,
   930→                _fullContent: isTruncated ? contextToSend : undefined,
   931→            })
   932→            this.reportContext = ''
   933→        }
   934→        if (!skipUserPush) {
   935→            this.messages.push({ role: 'user', content: text })
   936→            this.inputText = ''
   937→        }
   938→        this.$nextTick(this.scrollToBottom)
   939→
   940→        const assistantIdx = this.messages.length
   941→        this.messages.push({ role: 'assistant', content: '' })
   942→        this.isStreaming = true
   943→
   944→        const newContextChars = contextToSend ? contextToSend.length : 0
   945→        const limitAction = await this.checkContextOverLimit(newContextChars)
   946→        if (limitAction !== 'ok') {
   947→            this.messages.splice(msgStartIdx)
   948→            this.isStreaming = false
   949→            if (limitAction === 'newSession') { this.newSession() }
   950→            return
   951→        }
   952→
   953→        this.streamAbort = new AbortController()
   954→        try {
   955→            const history = this.messages.slice(0, -2).filter(m => m.role !== 'context' && !!m.content).slice(-10) as { role: 'user' | 'assistant'; content: string }[]
   956→            const processedContext = contextToSend ? compressDailyContext(contextToSend) : ''
   957→            if (processedContext) {
   958→                console.info('[AiChat] 压缩前上下文:\n', processedContext)
   959→            }
   960→            const compressedContext = processedContext ? await compressContext(processedContext) : ''
   961→            const response = await dailyReportAiApis.askAi({
   962→                role_id: this.currentRole!.id,
   963→                session_id: this.currentSessionId,
   964→                message: text,
   965→                report_context: compressedContext,
   966→                report_context_raw: contextToSend ? await compressContext(contextToSend) : undefined,
   967→                system_prompt: this.currentRole!.system_prompt
   968→                    + (sceneType && this.SCENE_OUTPUT_RULES[sceneType] || ''),
   969→                history,
   970→            }, this.streamAbort.signal)
   971→
   972→            const reader = response.body!.getReader()
   973→            const decoder = new TextDecoder()
   974→            let buffer = ''
   975→
   976→            while (true) {
   977→                const { done, value } = await reader.read()
   978→                if (done) {break}
   979→                buffer += decoder.decode(value, { stream: true })
   980→                const lines = buffer.split('\n')
   981→                buffer = lines.pop() ?? ''
   982→
   983→                for (const line of lines) {
   984→                    if (!line.startsWith('data: ')) {continue}
   985→                    try {
   986→                        const parsed = JSON.parse(line.slice(6))
   987→                        if (parsed.error_code !== 0) {
   988→                            this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   989→                            break
   990→                        }
   991→                        const data = parsed.data
   992→                        // TransferRequestService SSE 格式：meta（首帧，含 session_id）+ chat（流式内容）
   993→                        if (data.type === 'meta') {
   994→                            if (data.session_id && !this.currentSessionId) {
   995→                                this.currentSessionId = Number(data.session_id)
   996→                                this.loadSessions()
   997→                            }
   998→                        } else if (data.type === 'chat' && data.content) {
   999→                            this.messages[assistantIdx].content += data.content
  1000→                            this.$nextTick(this.scrollToBottom)
  1001→                        }
  1002→                    } catch {}
  1003→                }
  1004→            }
  1005→            // 流式读取完成，累加本次上下文字符数
  1006→            if (newContextChars > 0) {
  1007→                this.accumulatedContextChars += newContextChars
  1008→            }
  1009→        } catch (e: any) {
  1010→            if (e?.name !== 'AbortError') {
  1011→                console.error('[AiChat] sendAskMessage error:', e)
  1012→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
  1013→            }
  1014→        } finally {
  1015→            if (!this.messages[assistantIdx]?.content && this.streamAbort && !this.streamAbort.signal.aborted) {
  1016→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
  1017→            }
  1018→            this.streamAbort = null
  1019→            this.isStreaming = false
  1020→            // 落库：仅保存 AI 回复（context 和 user 消息已由后端 ask 接口保存，避免重复）
  1021→            const assistantContent = this.messages[assistantIdx]?.content
  1022→            if (assistantContent && this.currentSessionId) {
  1023→                await dailyReportAiApis.saveMessages({
  1024→                    role_id: this.currentRole!.id,
  1025→                    session_id: this.currentSessionId,
  1026→                    messages: [{ role: 'assistant', content: assistantContent }],
  1027→                })
  1028→            }
  1029→            await this.loadSessions()
  1030→        }
  1031→    }
  1032→
  1033→    // 加载更多 context 内容
  1034→    async loadMoreContext(m: { content: string; _id?: number; _truncated?: boolean; _fullLength?: number; _fullContent?: string; _loading?: boolean }) {
  1035→        if (!m._truncated || m._loading) {return}
  1036→        const chunkSize = 5000
  1037→
  1038→        // 实时场景：完整内容在本地 _fullContent 中，直接截取
  1039→        if (m._fullContent) {
  1040→            const offset = m.content.length
  1041→            const nextChunk = m._fullContent.slice(offset, offset + chunkSize)
  1042→            m.content += nextChunk
  1043→            m._truncated = (offset + chunkSize) < m._fullContent.length
  1044→            if (!m._truncated) {m._fullContent = undefined} // 加载完毕，释放内存
  1045→            return
  1046→        }
  1047→
  1048→        // 历史场景：从后端分段加载
  1049→        if (!m._id) {return}
  1050→        this.$set(m, '_loading', true)
  1051→        try {
  1052→            const offset = m.content.length
  1053→            const result = await dailyReportAiApis.getContextMore(m._id, offset)
  1054→            if (result) {
  1055→                m.content += result.content
  1056→                m._truncated = result.has_more
  1057→                m._fullLength = result.full_length
  1058→            }
  1059→        } finally {
  1060→            this.$set(m, '_loading', false)
  1061→        }
  1062→    }
