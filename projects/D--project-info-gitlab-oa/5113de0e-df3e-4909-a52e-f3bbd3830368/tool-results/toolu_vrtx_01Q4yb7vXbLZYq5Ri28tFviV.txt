   482→   482→    /** 检查累积上下文是否可能超限，返回 'ok' 继续发送 / 'newSession' 新建会话 / 'abort' 关闭弹窗不操作 */

   517→   517→    private async sendExtractSlots(text: string) {
   518→   518→        const limitAction = await this.checkContextOverLimit()
   519→   519→        if (limitAction === 'newSession') { this.newSession(); return }
   520→   520→        if (limitAction === 'abort') { return }
   521→   521→
   522→   522→        this.messages.push({ role: 'user', content: text })
   523→   523→        this.pendingSlotMessages.push({ role: 'user', content: text })
   524→   524→        this.inputText = ''
   525→   525→        this.$nextTick(this.scrollToBottom)
   526→   526→
   527→   527→        const assistantIdx = this.messages.length
   528→   528→        this.messages.push({ role: 'assistant', content: '' })
   529→   529→        this.isStreaming = true
   530→   530→
   531→   531→        try {
   532→   532→            // 收集可用分组名和所有成员花名
   533→   533→            const availableGroups = this.groupOptions.map(g => g.label)
   534→   534→            const availableMembers = [...new Set(this.groupOptions.flatMap(g => g.value))].filter(Boolean)
   535→   535→            // 续查模式：传入上次的槽位结果
   536→   536→            const previousContext = this.lastSlotResult ? {
   537→   537→                member_type: this.lastSlotResult.member_type,
   538→   538→                members: this.lastSlotResult.members,
   539→   539→                period_start: this.lastSlotResult.period_start,
   540→   540→                period_end: this.lastSlotResult.period_end,
   541→   541→            } : undefined
   542→   542→            const hasExistingContext = this.messages.some(m => m.role === 'context')
   543→   543→            const result = await dailyReportAiApis.extractSlots({
   544→   544→                message: text,
   545→   545→                conversation_id: this.slotsConversationId || undefined,
   546→   546→                has_existing_context: hasExistingContext || undefined,
   547→   547→                available_groups: availableGroups,
   548→   548→                available_members: availableMembers,
   549→   549→                previous_context: previousContext,
   550→   550→            })
   551→   551→
   552→   552→            if (!result) {
   553→   553→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   554→   554→                this.pendingSlotMessages.push({ role: 'assistant', content: this.messages[assistantIdx].content })
   555→   555→                await this.flushSlotMessages()
   556→   556→                return
   557→   557→            }
   558→   558→
   559→   559→            // 保存 conversation_id 用于多轮
   560→   560→            this.slotsConversationId = result.conversation_id
   561→   561→
   562→   562→            // 意图为分析已有数据 → 直接走问答，不重新取数据
   563→   563→            if (result.intent === 'analyze') {
   564→   564→                // 移除占位的 assistant 消息，sendAskMessage 会重新创建
   565→   565→                this.messages.splice(assistantIdx, 1)
   566→   566→                this.isStreaming = false
   567→   567→                await this.flushSlotMessages()
   568→   568→                await this.sendAskMessage(text, true)
   569→   569→                return
   570→   570→            }
   571→   571→
   572→   572→            if (!result.slots_complete) {
   573→   573→                // 槽位不完整，显示反问
   574→   574→                this.messages[assistantIdx].content = result.reply
   575→   575→                this.pendingSlotMessages.push({ role: 'assistant', content: result.reply })
   576→   576→                await this.flushSlotMessages()
   577→   577→                return
   578→   578→            }
   579→   579→
   580→   580→            // 槽位完整，保存结果用于续查
   581→   581→            this.lastSlotResult = result
   582→   582→            // 回显意图识别结果
   583→   583→            const nicknames = this.resolveMembers(result)
   584→   584→            const memberLabel = result.member_type === 'group'
   585→   585→                ? `分组：${result.members.join('、')}`
   586→   586→                : `成员：${result.members.join('、')}`
   587→   587→            const countLabel = nicknames.length >= 10 ? `，共 ${nicknames.length} 人` : ''
   588→   588→            const periodLabel = `${result.period_start} ~ ${result.period_end}`
   589→   589→            const slotSummary = `查询对象：${memberLabel}${countLabel}\n时间范围：${periodLabel}${result.period_raw ? `（${result.period_raw}）` : ''}`
   590→   590→            this.messages[assistantIdx].content = slotSummary + '\n\n正在同步日报数据...'
   591→   591→            this.pendingSlotMessages.push({ role: 'assistant', content: slotSummary })
   592→   592→            this.$nextTick(this.scrollToBottom)
   593→   593→
   594→   594→            if (nicknames.length === 0) {
   595→   595→                const errMsg = `未找到匹配的成员「${result.members.join('、')}」，请尝试使用花名或先手动同步数据。`
   596→   596→                this.messages[assistantIdx].content = errMsg
   597→   597→                this.pendingSlotMessages.push({ role: 'assistant', content: errMsg })
   598→   598→                await this.flushSlotMessages()
   599→   599→                return
   600→   600→            }
   601→   601→
   602→   602→            const context = await this.fetchReportContext(nicknames, result.period_start, result.period_end)
   603→   603→            if (!context) {
   604→   604→                this.messages[assistantIdx].content = '该时间段内没有找到日报数据。'
   605→   605→                this.pendingSlotMessages.push({ role: 'assistant', content: '该时间段内没有找到日报数据。' })
   606→   606→                await this.flushSlotMessages()
   607→   607→                return
   608→   608→            }
   609→   609→
   610→   610→            // 重置意图识别状态
   611→   611→            this.slotsConversationId = ''
   612→   612→
   613→   613→            // 意图识别消息先落库（会创建 session）
   614→   614→            await this.flushSlotMessages()
   615→   615→
   616→   616→            // 保留意图识别结果消息，走正常 ask 流程
   617→   617→            this.reportContext = context
   618→   618→            // 意图已由 extract-slots 处理，ask 只需分析同步好的日报数据
   619→   619→            await this.sendAskMessage('请根据以上日报数据进行分析')
   620→   620→        } catch {
   621→   621→            this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   622→   622→            this.pendingSlotMessages.push({ role: 'assistant', content: this.messages[assistantIdx].content })
   623→   623→            await this.flushSlotMessages()
   624→   624→        } finally {
   625→   625→            this.isStreaming = false
   626→   626→        }
   627→   627→    }

   632→   632→    private resolveMembers(result: ExtractSlotsResult): string[] {
   633→   633→        if (result.member_type === 'group') {
   634→   634→            // 匹配 groupOptions 的 label，展开为花名列表
   635→   635→            const matched = this.groupOptions.filter(g =>
   636→   636→                result.members.some(m => g.label.includes(m) || m.includes(g.label))
   637→   637→            )
   638→   638→            return [...new Set(matched.flatMap(g => g.value))].filter(Boolean)
   639→   639→        }
   640→   640→        // individual：LLM 返回的成员名必须在花名列表中匹配到，匹配不到则丢弃
   641→   641→        const allMembers = this.groupOptions.flatMap(g => g.value).filter(Boolean)
   642→   642→        return result.members
   643→   643→            .map(name => allMembers.find(m => m.includes(name) || name.includes(m)))
   644→   644→            .filter((n): n is string => !!n)
   645→   645→    }

   650→   650→    private async fetchReportContext(nicknames: string[], startDate: string, endDate: string): Promise<string> {
   651→   651→        const days = moment(endDate).diff(moment(startDate), 'day') + 1
   652→   652→        const perPage = Math.max(200, days + 10)
   653→   653→
   654→   654→        const results = await Promise.all(

   657→   657→                    start_date: startDate,
   658→   658→                    end_date: endDate,
   659→   659→                    nickname,
   660→   660→                    per_page: perPage,
   661→   661→                    page: 1,
   662→   662→                    show_rest: 1,
   663→   663→                    sys_site: this.sysSite,
   664→   664→                } as any)
   665→   665→            )
   666→   666→        )
   667→   667→
   668→   668→        const dateMap: Record<string, string[]> = {}
   669→   669→        results.forEach((res, idx) => {
   670→   670→            if (!res) {return}
   671→   671→            const name = nicknames[idx].replace(/\[.+]/, '')
   672→   672→            ;(res.reports ?? []).forEach((item: any) => {
   673→   673→                const date = item.date ?? ''
   674→   674→                if (!dateMap[date]) {dateMap[date] = []}
   675→   675→                if (item.report) {
   676→   676→                    dateMap[date].push(`${name}：\n${item.report.content ?? ''}`)
   677→   677→                } else if ([4, 5].includes(item.type_id)) {
   678→   678→                    dateMap[date].push(`${name}：休息日`)
   679→   679→                } else {
   680→   680→                    dateMap[date].push(`${name}：未提交`)
   681→   681→                }
   682→   682→            })
   683→   683→        })
   684→   684→
   685→   685→        const sortedDates = Object.keys(dateMap).sort((a, b) => b.localeCompare(a))
   686→   686→        return sortedDates
   687→   687→            .map(date => `【${date}】\n${dateMap[date].join('\n\n')}`)
   688→   688→            .join('\n\n---\n\n')
   689→   689→    }

   692→   692→     * 正常问答流程：带 reportContext 调 askAi（SSE 流式）
   693→   693→     */
   694→   694→    private async sendAskMessage(text: string, skipUserPush = false) {
   695→   695→        const contextToSend = this.reportContext
   696→   696→        // 记录本轮消息插入起点，取消时用于回退
   697→   697→        const msgStartIdx = this.messages.length
   698→   698→        if (contextToSend) {
   699→   699→            const contextLimit = 5000
   700→   700→            const isTruncated = contextToSend.length > contextLimit
   701→   701→            this.messages.push({
   702→   702→                role: 'context',
   703→   703→                content: isTruncated ? contextToSend.slice(0, contextLimit) : contextToSend,
   704→   704→                _truncated: isTruncated,
   705→   705→                _fullLength: contextToSend.length,
   706→   706→                _fullContent: isTruncated ? contextToSend : undefined,
   707→   707→            })
   708→   708→            this.reportContext = ''
   709→   709→        }
   710→   710→        if (!skipUserPush) {
   711→   711→            this.messages.push({ role: 'user', content: text })
   712→   712→            this.inputText = ''
   713→   713→        }
   714→   714→        this.$nextTick(this.scrollToBottom)
   715→   715→
   716→   716→        const assistantIdx = this.messages.length
   717→   717→        this.messages.push({ role: 'assistant', content: '' })
   718→   718→        this.isStreaming = true
   719→   719→
   720→   720→        const newContextChars = contextToSend ? contextToSend.length : 0
   721→   721→        const limitAction = await this.checkContextOverLimit(newContextChars)
   722→   722→        if (limitAction !== 'ok') {
   723→   723→            this.messages.splice(msgStartIdx)
   724→   724→            this.isStreaming = false
   725→   725→            if (limitAction === 'newSession') { this.newSession() }
   726→   726→            return
   727→   727→        }
   728→   728→
   729→   729→        this.streamAbort = new AbortController()
   730→   730→        try {
   731→   731→            const history = this.messages.slice(0, -2).filter(m => m.role !== 'context' && !!m.content).slice(-10) as { role: 'user' | 'assistant'; content: string }[]
   732→   732→            const processedContext = contextToSend ? compressDailyContext(contextToSend) : ''
   733→   733→            if (processedContext) {
   734→   734→                console.log('[AiChat] 压缩前上下文:\n', processedContext)
   735→   735→            }
   736→   736→            const compressedContext = processedContext ? await compressContext(processedContext) : ''
   737→   737→            const response = await dailyReportAiApis.askAi({
   738→   738→                role_id: this.currentRole!.id,
   739→   739→                session_id: this.currentSessionId,
   740→   740→                message: text,
   741→   741→                report_context: compressedContext,
   742→   742→                report_context_raw: contextToSend ? await compressContext(contextToSend) : undefined,
   743→   743→                system_prompt: this.currentRole!.system_prompt,
   744→   744→                history,
   745→   745→            }, this.streamAbort.signal)
   746→   746→
   747→   747→            const reader = response.body!.getReader()
   748→   748→            const decoder = new TextDecoder()
   749→   749→            let buffer = ''
   750→   750→
   751→   751→            while (true) {
   752→   752→                const { done, value } = await reader.read()
   753→   753→                if (done) {break}
   754→   754→                buffer += decoder.decode(value, { stream: true })
   755→   755→                const lines = buffer.split('\n')
   756→   756→                buffer = lines.pop() ?? ''
   757→   757→
   758→   758→                for (const line of lines) {
   759→   759→                    if (!line.startsWith('data: ')) {continue}
   760→   760→                    try {
   761→   761→                        const parsed = JSON.parse(line.slice(6))
   762→   762→                        if (parsed.error_code !== 0) {
   763→   763→                            this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   764→   764→                            break
   765→   765→                        }
   766→   766→                        const data = parsed.data
   767→   767→                        // TransferRequestService SSE 格式：meta（首帧，含 session_id）+ chat（流式内容）
   768→   768→                        if (data.type === 'meta') {
   769→   769→                            if (data.session_id && !this.currentSessionId) {
   770→   770→                                this.currentSessionId = Number(data.session_id)
   771→   771→                                this.loadSessions()
   772→   772→                            }
   773→   773→                        } else if (data.type === 'chat' && data.content) {
   774→   774→                            this.messages[assistantIdx].content += data.content
   775→   775→                            this.$nextTick(this.scrollToBottom)
   776→   776→                        }
   777→   777→                    } catch {}
   778→   778→                }
   779→   779→            }
   780→   780→            // 流式读取完成，累加本次上下文字符数
   781→   781→            if (newContextChars > 0) {
   782→   782→                this.accumulatedContextChars += newContextChars
   783→   783→            }
   784→   784→        } catch (e: any) {
   785→   785→            if (e?.name !== 'AbortError') {
   786→   786→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   787→   787→            }
   788→   788→        } finally {
   789→   789→            if (!this.messages[assistantIdx]?.content && this.streamAbort && !this.streamAbort.signal.aborted) {
   790→   790→                this.messages[assistantIdx].content = this.$t('aiChat.errorMsg') as string
   791→   791→            }
   792→   792→            this.streamAbort = null
   793→   793→            this.isStreaming = false
   794→   794→            // AI 回复落库：DB 存储现在由前端完成
   795→   795→            const finalContent = this.messages[assistantIdx]?.content
   796→   796→            if (finalContent && this.currentSessionId) {
   797→   797→                await dailyReportAiApis.saveMessages({
   798→   798→                    role_id: this.currentRole!.id,
   799→   799→                    session_id: this.currentSessionId,
   800→   800→                    messages: [{ role: 'assistant', content: finalContent }],
   801→   801→                })
   802→   802→            }
   803→   803→            await this.loadSessions()
   804→   804→        }
   805→   805→    }

   807→   807→    // 加载更多 context 内容
   808→   808→    async loadMoreContext(m: { content: string; _id?: number; _truncated?: boolean; _fullLength?: number; _fullContent?: string; _loading?: boolean }) {
   809→   809→        if (!m._truncated || m._loading) {return}
   810→   810→        const chunkSize = 5000
   811→   811→
   812→   812→        // 实时场景：完整内容在本地 _fullContent 中，直接截取
   813→   813→        if (m._fullContent) {
   814→   814→            const offset = m.content.length
   815→   815→            const nextChunk = m._fullContent.slice(offset, offset + chunkSize)
   816→   816→            m.content += nextChunk
   817→   817→            m._truncated = (offset + chunkSize) < m._fullContent.length
   818→   818→            if (!m._truncated) {m._fullContent = undefined} // 加载完毕，释放内存
   819→   819→            return
   820→   820→        }
   821→   821→
   822→   822→        // 历史场景：从后端分段加载
   823→   823→        if (!m._id) {return}
   824→   824→        this.$set(m, '_loading', true)
   825→   825→        try {
   826→   826→            const offset = m.content.length
   827→   827→            const result = await dailyReportAiApis.getContextMore(m._id, offset)
   828→   828→            if (result) {
   829→   829→                m.content += result.content
   830→   830→                m._truncated = result.has_more
   831→   831→                m._fullLength = result.full_length
   832→   832→            }
   833→   833→        } finally {
   834→   834→            this.$set(m, '_loading', false)
   835→   835→        }
   836→   836→    }
