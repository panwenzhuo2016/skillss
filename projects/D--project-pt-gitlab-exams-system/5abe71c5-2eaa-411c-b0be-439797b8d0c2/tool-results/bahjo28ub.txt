+function applyHighlights(el, anns) {

+function highlightTextInNode(root, text, annId) {
+  if (!text) return;
+  const walker = document.createTreeWalker(root, NodeFilter.SHOW_TEXT);
+  const matches = [];
+  while (walker.nextNode()) {
+    const node = walker.currentNode;
+    const idx = node.textContent.indexOf(text);
+    if (idx !== -1) { matches.push({ node, idx }); }
+  }
+  // 从后往前替换，避免偏移
+  for (let i = matches.length - 1; i >= 0; i--) {
+    const { node, idx } = matches[i];
+    const before = node.textContent.substring(0, idx);
+    const match = node.textContent.substring(idx, idx + text.length);
+    const after = node.textContent.substring(idx + text.length);
+    const span = document.createElement('span');
+    span.className = 'ann-highlight';
+    span.dataset.annId = annId;
+    span.textContent = match;
+    span.title = '';
+    span.onmouseenter = function(e) { showAnnotationForHighlight(annId, e); };
+    span.onmouseleave = function(e) { scheduleHideBubble(); };
+    const parent = node.parentNode;
+    if (after) parent.insertBefore(document.createTextNode(after), node.nextSibling);
+    parent.insertBefore(span, node.nextSibling);
+    node.textContent = before;
+  }

+  }
+
+  const data = await resp.json();
+  if (data.code !== 200) { throw new Error(data.message || '请求失败'); }

+    const m = first.id.match(/^(.*?\d+)([A-Za-z])$/);
+    if (m) {
+      prefix = m[1]; suffixType = 'alpha';
+    } else {
+      const m2 = first.id.match(/^([A-Za-z]*)(\d+)$/);

+    } else {
+      sc.id = prefix + String(i + 1).padStart(2, '0'); // 01, 02, 03...
+    }
+  });

+// ═══════════════════════════════════════════════════════════════════════
+// PARSER: Markdown → 结构化数据
+// ═══════════════════════════════════════════════════════════════════════
+function parseOutline(text) {
+  const lines = text.replace(/\r\n/g,'\n').replace(/\r/g,'\n').split('\n');
+  const result = [];
+  let act=null, chapter=null, scene=null;
+  let summaryLines=[], bodyLines=[], detailLines=[], actNoteLines=[], inDetail=false;
+
+  function flush() {
+    if (scene) {
+      scene.summary = summaryLines.join('\n').trimEnd();
+      scene.body    = bodyLines.join('\n').trimEnd();
+      scene.detail  = detailLines.join('\n').trim();
+    }
+    if (act && !chapter) {
+      act.notes = actNoteLines.join('\n').trim();
+    }
+    summaryLines=[]; bodyLines=[]; detailLines=[]; actNoteLines=[]; inDetail=false; scene=null;
+  }
+
+  for (const raw of lines) {
+    const line = raw.trim();
+    if (line.startsWith('## ')&&!line.startsWith('### ')) {
+      flush(); act={title:line.slice(3).trim(),chapters:[],open:true,notes:''}; chapter=null; result.push(act); continue;
+    }
+    if (line.startsWith('### ')) {
+      if (act && !chapter) { act.notes = actNoteLines.join('\n').trim(); actNoteLines=[]; }
+      flush(); if(act){chapter={title:line.slice(4).trim(),scenes:[],open:true};act.chapters.push(chapter);} continue;
+    }
+    const sm=line.match(/^\*\*(\d+[A-Za-z]+(?:\.\d+)?)[. ]\s*(.+?)\*\*/);
+    if (sm&&chapter) {
+      flush(); scene={id:sm[1],title:sm[2].trim(),summary:'',body:'',detail:''}; chapter.scenes.push(scene); continue;
+    }
+    if (act && !chapter && !scene) { actNoteLines.push(raw); continue; }
+    if (!scene) continue;
+    if (line.startsWith('<details>')) { inDetail=true; continue; }
+    if (line==='</details>') { inDetail=false; continue; }
+    if (inDetail) { detailLines.push(raw); continue; }
+    if (raw.startsWith('> ') || raw==='>') {
+      summaryLines.push(raw.startsWith('> ') ? raw.slice(2) : ''); continue;
+    }
+    if (!line.startsWith('#') && !line.startsWith('```')) bodyLines.push(raw);
+  }
+  flush(); return result;
+}

+// ═══════════════════════════════════════════════════════════════════════
+// SERIALIZE: 结构化数据 → Markdown
+// ═══════════════════════════════════════════════════════════════════════
+function serializeOutline() {
+  const lines=[];
+  if(fileTitle) lines.push(fileTitle,'');
+  acts.forEach(act=>{
+    lines.push('## '+act.title,'');
+    if(act.notes) lines.push(act.notes,'');
+    act.chapters.forEach(ch=>{
+      lines.push('### '+ch.title,'');
+      ch.scenes.forEach(sc=>{
+        lines.push('**'+sc.id+'. '+sc.title+'**');
+        if(sc.summary){
+          sc.summary.split('\n').forEach(l=>lines.push('> '+l));
+        } else if(sc.body){
+          lines.push(sc.body);
+        }
+        if(sc.detail){
+          lines.push('','<details><summary>场景细节</summary>','',sc.detail,'','</details>');
+        }
+        lines.push('');
+      });
+    });
+  });
+  return lines.join('\n');
+}

+// 对比两个版本的场景差异
+function diffScenes(scenesA, scenesB) {
+  const idsA = new Set(Object.keys(scenesA));
+  const idsB = new Set(Object.keys(scenesB));
+  const added = [], deleted = [], modified = [];
+  // B 中有但 A 中没有 → 新增场景
+  for (const id of idsB) {
+    if (!idsA.has(id)) added.push({ id, title: scenesB[id].title });
+  }
+  // A 中有但 B 中没有 → 已删除场景
+  for (const id of idsA) {
+    if (!idsB.has(id)) deleted.push({ id, title: scenesA[id].title });
+  }
+  // 两边都有 → 对比内容是否变化
+  for (const id of idsA) {
+    if (!idsB.has(id)) continue;
+    const a = scenesA[id], b = scenesB[id];
+    if (a.title !== b.title || a.text !== b.text || a.detail !== b.detail) {
+      modified.push({ id, titleA: a.title, titleB: b.title, textA: a.text, textB: b.text, detailA: a.detail, detailB: b.detail });
+    }
+  }
+  return { added, deleted, modified };

+function lcsMatrix(a, b) {

+  let prev = new Array(n + 1).fill(0);
+  let curr = new Array(n + 1).fill(0);
+  for (let i = 1; i <= m; i++) {
+    for (let j = 1; j <= n; j++) {
+      if (a[i - 1] === b[j - 1]) curr[j] = prev[j - 1] + 1;
+      else curr[j] = Math.max(prev[j], curr[j - 1]);
+    }
+    [prev, curr] = [curr, prev.fill(0)];
+  }
+  return prev;
+}

+// 完整 LCS 回溯（用全表，文本不会太长）
+function diffTokens(oldTokens, newTokens) {
+  const m = oldTokens.length, n = newTokens.length;
+  // 构建完整 DP 表
+  const dp = [];
+  for (let i = 0; i <= m; i++) {
+    dp[i] = new Array(n + 1).fill(0);
+  }
+  for (let i = 1; i <= m; i++) {
+    for (let j = 1; j <= n; j++) {
+      if (oldTokens[i - 1] === newTokens[j - 1]) dp[i][j] = dp[i - 1][j - 1] + 1;
+      else dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
+    }
+  }
+  // 回溯生成 ops: 'eq' | 'del' | 'ins'
+  const ops = [];
+  let i = m, j = n;
+  while (i > 0 || j > 0) {
+    if (i > 0 && j > 0 && oldTokens[i - 1] === newTokens[j - 1]) {
+      ops.push({ type: 'eq', text: oldTokens[i - 1] });
+      i--; j--;
+    } else if (j > 0 && (i === 0 || dp[i][j - 1] >= dp[i - 1][j])) {
+      ops.push({ type: 'ins', text: newTokens[j - 1] });
+      j--;
+    } else {
+      ops.push({ type: 'del', text: oldTokens[i - 1] });
+      i--;
+    }
+  }
+  ops.reverse();
+  return ops;
+}

+function renderDiffHtml(oldText, newText) {
+  if (!oldText && !newText) return '';
+  if (oldText === newText) return esc(newText);
+  if (!oldText) return '<ins class="diff-ins">' + esc(newText) + '</ins>';
+  if (!newText) return '<del class="diff-del">' + esc(oldText) + '</del>';

+  const ops = diffTokens(tokenize(oldText), tokenize(newText));
+  let html = '';
+  let i = 0;
+  while (i < ops.length) {
+    const op = ops[i];
+    if (op.type === 'eq') {
+      html += esc(op.text);
+      i++;
+    } else if (op.type === 'del') {
+      // 收集连续 del，后面可能紧跟连续 ins（构成替换）
+      let delText = '';
+      while (i < ops.length && ops[i].type === 'del') { delText += ops[i].text; i++; }
+      let insText = '';
+      while (i < ops.length && ops[i].type === 'ins') { insText += ops[i].text; i++; }
+      if (insText) {

+        html += '<del class="diff-del">' + esc(delText) + '</del>';

+      } else {

+    } else if (op.type === 'ins') {
