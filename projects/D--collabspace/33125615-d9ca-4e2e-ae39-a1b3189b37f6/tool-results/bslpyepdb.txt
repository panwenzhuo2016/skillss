+            log.warn("[1822-Approval] resolveRequestId: getNotificationExtras returned empty notificationId={} approverUserId={}",
+                    notificationId, approverUserId);
+            throw new BusinessException(
+                com.yottastudios.ns.collabspace.control.enums.PermissionRequestException.NOTIFICATION_NOT_VALID);
+        }
+        String templateId = extras.get("templateId");

+            log.warn("[1822-Approval] resolveRequestId: not a permission request notification notificationId={} templateId={}",
+                    notificationId, templateId);
+            throw new BusinessException(
+                com.yottastudios.ns.collabspace.control.enums.PermissionRequestException.NOTIFICATION_NOT_VALID);
+        }
+        String requestId = extras.get("requestId");
+        if (requestId == null || requestId.isBlank()) {
+            log.warn("[1822-Approval] resolveRequestId: requestId missing in extras notificationId={}",
+                    notificationId);
+            throw new BusinessException(
+                com.yottastudios.ns.collabspace.control.enums.PermissionRequestException.NOTIFICATION_NOT_VALID);
+        }
+        log.info("[1822-Approval] resolveRequestId ok notificationId={} requestId={}", notificationId, requestId);
+        return requestId;
+    }

+    @Override
+    @Transactional(rollbackFor = Exception.class)
+    public PermissionRequestVo submitRequest(String documentId, DocumentPermissionRequestRo ro) {
+        Long applicantUserId = SessionContext.getUserId();
+        ExceptionUtil.isNotNull(applicantUserId, PermissionRequestException.NODE_ACCESS_DENIED);
+        ExceptionUtil.isNotNull(documentId, PermissionRequestException.NODE_NOT_EXIST);
+
+        // 1. Check node exists and get spaceId; resolve numeric pk to appSpaceNodeId
+        SpaceNodeEntity spaceNode = getSpaceNodeByNodeId(documentId);
+        String resolvedDocumentId = spaceNode.getAppSpaceNodeId();
+        String spaceId = spaceNode.getSpaceId();
+        String documentName = spaceNode.getName();
+
+        // 2. Check user is in space
+        boolean isInSpace = spaceQueryManager.isUserInSpace(applicantUserId, spaceId);
+        ExceptionUtil.isTrue(isInSpace, PermissionRequestException.MEMBER_NOT_IN_SPACE);
+
+        // 3. Check no pending request for same document
+        PermissionRequestEntity pending = baseMapper.selectPendingByDocumentAndApplicant(resolvedDocumentId, applicantUserId);
+        ExceptionUtil.isNull(pending, PermissionRequestException.ALREADY_SUBMITTED_REQUEST);
+
+        // 4. Check user does not already have node permission
+        boolean hasPermission = checkUserHasNodePermission(spaceId, applicantUserId, resolvedDocumentId);
+        ExceptionUtil.isFalse(hasPermission, PermissionRequestException.ALREADY_HAVE_PERMISSION);
+
+        // 5. Debounce: short period repeated submissions only processed once
+        String debounceKey = REDIS_DEBOUNCE_PREFIX + applicantUserId + ":" + resolvedDocumentId;
+        Boolean firstSubmit = redisTemplate.opsForValue().setIfAbsent(debounceKey, "1", DEBOUNCE_SECONDS, TimeUnit.SECONDS);
+        if (Boolean.FALSE.equals(firstSubmit)) {
+            log.info("[1822-PermApply] debounce applicantUserId={} documentId={}", applicantUserId, documentId);
+            throw new BusinessException(PermissionRequestException.ALREADY_SUBMITTED_REQUEST);
+        }
+
+        // 6. Create permission request record
+        String requestId = REQUEST_ID_PREFIX + IdUtil.getSnowflakeNextId();
+        PermissionRequestEntity entity = PermissionRequestEntity.builder()
+            .id(requestId)
+            .applicantUserId(applicantUserId)
+            .documentId(resolvedDocumentId)
+            .documentName(documentName)
+            .spaceId(spaceId)
+            .requestedPermissionType(ro.getRequestedPermissionType())
+            .status(STATUS_PENDING)
+            .isDeleted(0)
+            .createdAt(LocalDateTime.now())
+            .updatedAt(LocalDateTime.now())
+            .build();
+
+        boolean flag = save(entity);
+        if (!flag) {
+            log.error("[1822-PermApply] ERROR CREATE insert failed applicantUserId={} documentId={} spaceId={}",
+                applicantUserId, resolvedDocumentId, spaceId);
+            throw new BusinessException(PermissionRequestException.INSERT_ERROR);
+        }
+        log.info("[1822-PermApply] CREATE request requestId={} applicantUserId={} documentId={} spaceId={} requestedType={}",
+            requestId, applicantUserId, resolvedDocumentId, spaceId, ro.getRequestedPermissionType());
+
+        // 7. Get space name and applicant nickname
+        SpaceEntity space = spaceService.getBySpaceId(spaceId);
+        String spaceName = space != null && space.getName() != null ? space.getName() : "";
+        String applicantNickName = getApplicantNickName(applicantUserId);
+
+        // 8. Notify admins & applicant via apitable gRPC (async)
+        notifyAdminsAsync(entity, documentName, applicantNickName, spaceName);
+        notifyApplicantAsync(entity, documentName, spaceName);
+
+        return buildVo(entity, documentName, spaceName);
+    }

+    private SpaceNodeEntity getSpaceNodeByNodeId(String nodeId) {
+        ExceptionUtil.isNotNull(nodeId, PermissionRequestException.NODE_NOT_EXIST);
+        // Always try appSpaceNodeId first (including deleted rows): OnlyOffice nodes use a numeric
+        // fileId as appSpaceNodeId, which must NOT be treated as the collabspace pk.
+        List<SpaceNodeEntity> byAppId = spaceNodeService.listByAppSpaceNodeIdsAllStatus(Collections.singletonList(nodeId));
+        if (CollUtil.isNotEmpty(byAppId)) return byAppId.get(0);
+        // Fallback: treat as collabspace pk if numeric (non-OnlyOffice node sent its own pk)
+        if (nodeId.matches("^\\d+$")) {
+            SpaceNodeEntity node = spaceNodeService.getByIdAllStatus(Long.parseLong(nodeId));
+            ExceptionUtil.isNotNull(node, PermissionRequestException.NODE_NOT_EXIST);
+            return node;
+        }
+        ExceptionUtil.isNotNull(null, PermissionRequestException.NODE_NOT_EXIST);
+        return null; // unreachable
+    }

+    private SpaceNodeEntity getSpaceNodeByNodeIdSafe(String nodeId) {
+        if (nodeId == null) return null;
+        // Always try by appSpaceNodeId column first (including deleted rows).
+        List<SpaceNodeEntity> nodes = spaceNodeService.listByAppSpaceNodeIdsAllStatus(Collections.singletonList(nodeId));

+        if (nodeId.matches("^\\d+$")) {
